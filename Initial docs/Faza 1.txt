RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_companies_updated BEFORE UPDATE ON companies
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER trg_agents_updated BEFORE UPDATE ON agents
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

import pg from 'pg';
import { env } from '../env.js';
import { logger } from '../utils/logger.js';

const pool = new pg.Pool({
  connectionString: env.DATABASE_URL,
  max: 20,
  idleTimeoutMillis: 30000,
});

pool.on('error', (err) => {
  logger.error({ err }, 'Unexpected database pool error');
});

export const db = {
  query: <T extends pg.QueryResultRow = any>(
    text: string,
    params?: any[]
  ) => pool.query<T>(text, params),

  getClient: () => pool.connect(),

  // Transakcja z auto-rollback
  transaction: async <T>(
    fn: (client: pg.PoolClient) => Promise<T>
  ): Promise<T> => {
    const client = await pool.connect();
    try {
      await client.query('BEGIN');
      const result = await fn(client);
      await client.query('COMMIT');
      return result;
    } catch (err) {
      await client.query('ROLLBACK');
      throw err;
    } finally {
      client.release();
    }
  },

  close: () => pool.end(),
};

import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';
import { db } from './client.js';
import { logger } from '../utils/logger.js';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

async function migrate() {
  const schema = fs.readFileSync(
    path.join(__dirname, 'schema.sql'),
    'utf-8'
  );

  try {
    await db.query(schema);
    logger.info('Database migration completed successfully');
  } catch (err: any) {
    // Jeśli tabele już istnieją, to OK
    if (err.code === '42P07') {
      logger.info('Tables already exist, skipping migration');
    } else {
      logger.error({ err }, 'Migration failed');
      throw err;
    }
  } finally {
    await db.close();
  }
}

import Anthropic from '@anthropic-ai/sdk';
import { env } from '../env.js';
import { logger } from '../utils/logger.js';
import type { AgentRuntime, AgentContext, AgentResponse, AgentAction } from './types.js';

const COST_PER_MTK_INPUT = 3.0;   // Claude Sonnet — $3/MTk input
const COST_PER_MTK_OUTPUT = 15.0;  // $15/MTk output

export class ClaudeRuntime implements AgentRuntime {
  name = 'claude';
  private client: Anthropic;

  constructor() {
    if (!env.ANTHROPIC_API_KEY) {
      throw new Error('ANTHROPIC_API_KEY is required for Claude runtime');
    }
    this.client = new Anthropic({ apiKey: env.ANTHROPIC_API_KEY });
  }

  async execute(context: AgentContext): Promise<AgentResponse> {
    const actionSchema = `
Respond with a JSON object containing your actions. Format:
{
  "thinking": "your reasoning about what to do",
  "actions": [
    {
      "type": "complete_task",
      "result": "description of what you accomplished"
    }
  ]
}

Available action types:
- complete_task: { "type": "complete_task", "result": "..." }
- delegate: { "type": "delegate", "to": "agent_role", "description": "..." }
- message: { "type": "message", "to": "agent_role", "content": "..." }
- blocked: { "type": "blocked", "reason": "..." }
- continue: { "type": "continue", "result": "progress so far..." }

You may include multiple actions. Always include at least one.
Respond ONLY with valid JSON.`;

    const messages: Anthropic.MessageParam[] = [
      ...context.conversationHistory.map((m) => ({
        role: m.role as 'user' | 'assistant',
        content: m.content,
      })),
      {
        role: 'user' as const,
        content: `${context.taskDescription}\n\n${actionSchema}`,
      },
    ];

    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 4096,
      system: context.systemPrompt,
      messages,
    });

    const rawText =
      response.content[0].type === 'text' ? response.content[0].text : '';

    // Parse response
    let actions: AgentAction[] = [];
    try {
      const jsonMatch = rawText.match(/\{[\s\S]*\}/);
      if (jsonMatch) {
        const parsed = JSON.parse(jsonMatch[0]);
        actions = parsed.actions || [];
      }
    } catch (err) {
      logger.warn({ rawText }, 'Failed to parse agent JSON, treating as complete');
      actions = [{ type: 'complete_task', result: rawText }];
    }

    const inputTokens = response.usage.input_tokens;
    const outputTokens = response.usage.output_tokens;
    const costUsd =
      (inputTokens / 1_000_000) * COST_PER_MTK_INPUT +
      (outputTokens / 1_000_000) * COST_PER_MTK_OUTPUT;

    return {
      actions,
      rawResponse: rawText,
      cost: {
        input_tokens: inputTokens,
        output_tokens: outputTokens,
        usd: costUsd,
      },
    };
  }
}

import OpenAI from 'openai';
import { env } from '../env.js';
import { logger } from '../utils/logger.js';
import type { AgentRuntime, AgentContext, AgentResponse, AgentAction } from './types.js';

const COST_PER_MTK_INPUT = 2.5;   // GPT-4o
const COST_PER_MTK_OUTPUT = 10.0;

export class OpenAIRuntime implements AgentRuntime {
  name = 'openai';
  private client: OpenAI;

  constructor() {
    if (!env.OPENAI_API_KEY) {
      throw new Error('OPENAI_API_KEY is required for OpenAI runtime');
    }
    this.client = new OpenAI({ apiKey: env.OPENAI_API_KEY });
  }

  async execute(context: AgentContext): Promise<AgentResponse> {
    const actionSchema = `
Respond with a JSON object containing your actions. Format:
{
  "thinking": "your reasoning",
  "actions": [
    { "type": "complete_task", "result": "..." }
  ]
}

Available action types:
- complete_task: { "type": "complete_task", "result": "..." }
- delegate: { "type": "delegate", "to": "agent_role", "description": "..." }
- message: { "type": "message", "to": "agent_role", "content": "..." }
- blocked: { "type": "blocked", "reason": "..." }

Respond ONLY with valid JSON.`;

    const messages: OpenAI.ChatCompletionMessageParam[] = [

      ...context.conversationHistory.map((m) => ({

    let actions: AgentAction[] = [];

      actions = parsed.actions || [];

      logger.warn({ rawText }, 'Failed to parse agent JSON');
      actions = [{ type: 'complete_task', result: rawText }];
    }

    const usage = response.usage;
    const inputTokens = usage?.prompt_tokens || 0;
    const outputTokens = usage?.completion_tokens || 0;
    const costUsd =
      (inputTokens / 1_000_000) * COST_PER_MTK_INPUT +
      (outputTokens / 1_000_000) * COST_PER_MTK_OUTPUT;

    return {
      actions,
      rawResponse: rawText,
      cost: {
        input_tokens: inputTokens,
        output_tokens: outputTokens,
        usd: costUsd,
      },
    };
  }

src/runtimes/registry.ts
TypeScript
import type { AgentRuntime } from './types.js';
import { ClaudeRuntime } from './claude.js';
import { OpenAIRuntime } from './openai.js';
import { env } from '../env.js';
import { logger } from '../utils/logger.js';

const runtimes: Map<string, AgentRuntime> = new Map();

export function initRuntimes() {
  if (env.ANTHROPIC_API_KEY) {
    runtimes.set('claude', new ClaudeRuntime());
    logger.info('Registered Claude runtime');
  }

  if (env.OPENAI_API_KEY) {
    runtimes.set('openai', new OpenAIRuntime());
    logger.info('Registered OpenAI runtime');
  }

  if (runtimes.size === 0) {
    logger.warn('No API keys configured — no runtimes available');
  }
}

export function getRuntime(name: string): AgentRuntime {
  const runtime = runtimes.get(name);
  if (!runtime) {
    throw new Error(
      `Runtime "${name}" not available. Configured: ${[...runtimes.keys()].join(', ')}`
    );
  }
  return runtime;
}

export function listRuntimes(): string[] {
  return [...runtimes.keys()];
}

import { db } from '../db/client.js';
import { logger } from '../utils/logger.js';

interface TaskRow {
  id: string;
  title: string;
  description: string;
  goal_id: string;
  status: string;
  priority: number;
}

/**
 * Atomically checkout the next available task for an agent.
 * Uses FOR UPDATE SKIP LOCKED to prevent double-checkout.
 */
export async function checkoutNextTask(
  agentId: string,
  companyId: string
): Promise<TaskRow | null> {
  const result = await db.query<TaskRow>(
    `
    UPDATE tasks
    SET
      locked_by = $1,
      locked_at = now(),
      status = 'in_progress'
    WHERE id = (
      SELECT id FROM tasks
      WHERE company_id = $2
        AND assigned_to = $1
        AND status = 'assigned'
        AND locked_by IS NULL
      ORDER BY priority DESC, created_at ASC
      LIMIT 1
      FOR UPDATE SKIP LOCKED
    )
    RETURNING *
    `,
    [agentId, companyId]
  );

  if (result.rows.length === 0) {
    return null;
  }

  logger.debug(
    { agentId, taskId: result.rows[0].id },
    'Task checked out'
  );

  return result.rows[0];
}

/**
 * Get the currently active (in_progress + locked) task for an agent.
 */
export async function getActiveTask(
  agentId: string
): Promise<TaskRow | null> {
  const result = await db.query<TaskRow>(
    `
    SELECT * FROM tasks
    WHERE assigned_to = $1
      AND status = 'in_progress'
      AND locked_by = $1
    ORDER BY priority DESC
    LIMIT 1
    `,
    [agentId]
  );

  return result.rows[0] || null;
}

/**
 * Release a task lock (on completion or error).
 */
export async function releaseTask(
  taskId: string,
  status: string,
  result?: string
) {
  await db.query(
    `
    UPDATE tasks
    SET
      locked_by = NULL,
      locked_at = NULL,
      status = $2,
      result = COALESCE($3, result)
    WHERE id = $1
    `,
    [taskId, status, result]
  );
}

interface GoalRow {

interface AgentRow {
  id: string;
  name: string;
  role: string;
  title: string;
  reports_to: string | null;
  system_prompt: string | null;
}

interface MessageRow {
  from_agent: string | null;
  from_name?: string;
  content: string;
  type: string;
  created_at: Date;
}

/**
 * Walk up the goal tree to build full ancestry.
 */
async function getGoalAncestry(goalId: string | null): Promise<GoalRow[]> {
  if (!goalId) return [];

  const chain: GoalRow[] = [];
  let currentId: string | null = goalId;

  while (currentId) {
    const result = await db.query<GoalRow>(
      'SELECT * FROM goals WHERE id = $1',
      [currentId]
    );
    if (result.rows.length === 0) break;
    chain.unshift(result.rows[0]); // parent first
    currentId = result.rows[0].parent_id;
  }

  return chain;
}

/**

async function getTaskMessages(taskId: string): Promise<MessageRow[]> {

}

/**
 * Build full context for an agent executing a task.
 */
export async function buildContext(
  agent: AgentRow,
  task: any,
  companyMission: string
) {
  // Goal ancestry
  const goalChain = await getGoalAncestry(task.goal_id);

  // Boss
  let boss: AgentRow | null = null;
  if (agent.reports_to) {
    const r = await db.query<AgentRow>(
      'SELECT * FROM agents WHERE id = $1',
      [agent.reports_to]
    );
    boss = r.rows[0] || null;
  }

  // Direct reports
  const reportsResult = await db.query<AgentRow>(
    'SELECT * FROM agents WHERE reports_to = $1 AND status != $2',
    [agent.id, 'terminated']
  );
  const reports = reportsResult.rows;

  // Conversation history
  const messages = await getTaskMessages(task.id);

  // Session state
  const sessionResult = await db.query(
    'SELECT state FROM agent_sessions WHERE agent_id = $1 AND task_id = $2',
    [agent.id, task.id]
  );
  const sessionState = sessionResult.rows[0]?.state || null;

  // Build system prompt
  const systemPrompt = `
${agent.system_prompt || `You are ${agent.name}, a skilled ${agent.role}.`}

COMPANY MISSION: ${companyMission}

YOUR IDENTITY:
- Name: ${agent.name}
- Role: ${agent.role}
- Title: ${agent.title || agent.role}
- Reports to: ${boss ? `${boss.name} (${boss.role})` : 'Board (human oversight)'}
- Direct reports: ${reports.length > 0 ? reports.map((r) => `${r.name} (${r.role})`).join(', ') : 'none'}

${
  goalChain.length > 0
    ? `GOAL HIERARCHY (why this task exists):
${goalChain.map((g, i) => '  '.repeat(i) + '→ ' + g.title + (g.description ? `: ${g.description}` : '')).join('\n')}`
    : ''
}

RULES:
1. Always respond with valid JSON containing your actions.
2. Focus on your assigned task. Don't exceed your role.
3. If you need something from another agent, use the "message" action.
4. If you can't proceed, use "blocked" with a clear reason.
5. If work is partially done, use "continue" to save progress.
6. When the task is fully complete, use "complete_task" with a thorough result.
`.trim();

  // Build conversation history for the LLM
  const conversationHistory = messages.map((m) => ({
    role: (m.from_agent === agent.id ? 'assistant' : 'user') as
      | 'user'
      | 'assistant',
    content: `[${m.from_name || 'Board'}] (${m.type}): ${m.content}`,
  }));

  const taskDescription = `
CURRENT TASK:
Title: ${task.title}
Description: ${task.description || 'No additional description.'}
Status: ${task.status}
Priority: ${task.priority}

${sessionState ? `PREVIOUS SESSION STATE:\n${JSON.stringify(sessionState, null, 2)}` : 'This is a fresh task — no previous work.'}

What will you do?
`.trim();

  return {
    systemPrompt,
    taskDescription,
    conversationHistory,
    availableActions: [
      'complete_task',
      'delegate',
      'message',
      'blocked',
      'continue',
    ],
    sessionState,
  };
}

/**
 * Run a single heartbeat for one agent.
 */
export async function agentHeartbeat(agent: AgentRow): Promise<void> {
  const startTime = Date.now();
  logger.info({ agent: agent.name, agentId: agent.id }, '💓 Heartbeat start');

  try {
    // 1. Check budget
    const currentMonth = new Date().toISOString().slice(0, 7) + '-01';
    const budgetResult = await db.query(
      'SELECT * FROM budgets WHERE agent_id = $1 AND month = $2',
      [agent.id, currentMonth]
    );
    const budget = budgetResult.rows[0];

    if (budget && parseFloat(budget.spent_usd) >= parseFloat(budget.limit_usd)) {
      logger.warn({ agent: agent.name }, '💰 Budget exceeded, skipping');
      await logHeartbeat(agent.id, null, 'budget_exceeded', Date.now() - startTime, 0);
      return;
    }

    // 2. Find work
    let task = await getActiveTask(agent.id);
    if (!task) {
      task = await checkoutNextTask(agent.id, agent.company_id);
    }
    if (!task) {
      logger.debug({ agent: agent.name }, '😴 No tasks, idle');
      await logHeartbeat(agent.id, null, 'idle', Date.now() - startTime, 0);
      return;
    }

    // 3. Update agent status
    await db.query(
      "UPDATE agents SET status = 'working' WHERE id = $1",
      [agent.id]
    );

    // 4. Get company mission
    const companyResult = await db.query(
      'SELECT mission FROM companies WHERE id = $1',
      [agent.company_id]
    );
    const mission = companyResult.rows[0]?.mission || 'No mission defined.';

    // 5. Build context
    const context = await buildContext(agent, task, mission);

    // 6. Execute
    const runtime = getRuntime(agent.runtime);
    const response = await runtime.execute(context);

    logger.info(
      {
        agent: agent.name,
        task: task.title,
        actions: response.actions.map((a) => a.type),
        cost: response.cost.usd.toFixed(4),
      },
      '🤖 Agent responded'
    );

    // 7. Record cost
    if (budget) {
      await db.query(
        'UPDATE budgets SET spent_usd = spent_usd + $1 WHERE id = $2',
        [response.cost.usd, budget.id]
      );
    }

    // 8. Log the raw response as a message
    await db.query(
      `INSERT INTO messages (company_id, task_id, from_agent, content, type, metadata)
       VALUES ($1, $2, $3, $4, 'heartbeat_log', $5)`,
      [
        agent.company_id,
        task.id,
        agent.id,
        response.rawResponse,
        JSON.stringify({ cost: response.cost }),
      ]
    );

    // 9. Process actions
    for (const action of response.actions) {
      await processAction(agent, task, action);
    }

    // 10. Update agent status back to idle
    await db.query(
      "UPDATE agents SET status = 'idle' WHERE id = $1",
      [agent.id]
    );

    // 11. Log heartbeat
    await logHeartbeat(
      agent.id,
      task.id,
      'worked',
      Date.now() - startTime,
      response.cost.usd
    );

    // 12. Audit
    await db.query(
      `INSERT INTO audit_log (company_id, agent_id, action, entity_type, entity_id, details, cost_usd)
       VALUES ($1, $2, 'heartbeat.completed', 'task', $3, $4, $5)`,
      [
        agent.company_id,
        agent.id,
        task.id,
        JSON.stringify({
          actions: response.actions.map((a) => a.type),
          tokens: response.cost,
        }),
        response.cost.usd,
      ]
    );
  } catch (err) {
    logger.error({ err, agent: agent.name }, '❌ Heartbeat failed');
    await db.query(
      "UPDATE agents SET status = 'idle' WHERE id = $1",
      [agent.id]
    );
    await logHeartbeat(agent.id, null, 'error', Date.now() - startTime, 0, err);
  }
}

/**
 * Process a single action from an agent's response.
 */
async function processAction(
  agent: AgentRow,
  task: any,
  action: AgentAction
): Promise<void> {
  switch (action.type) {
    case 'complete_task':
      await releaseTask(task.id, 'done', action.result);
      logger.info({ agent: agent.name, task: task.title }, '✅ Task completed');
      break;

    case 'delegate': {
      // Find target agent by role
      const targetResult = await db.query(
        `SELECT id FROM agents
         WHERE company_id = $1
           AND role = $2
           AND status != 'terminated'
         LIMIT 1`,
        [agent.company_id, action.to]
      );
      const targetId = targetResult.rows[0]?.id;

      if (targetId) {
        await db.query(
          `INSERT INTO tasks
            (company_id, goal_id, parent_id, title, description, assigned_to, created_by, status, priority)
           VALUES ($1, $2, $3, $4, $5, $6, $7, 'assigned', $8)`,
          [
            agent.company_id,
            task.goal_id,
            task.id,
            action.description || `Subtask from ${agent.name}`,
            action.description,
            targetId,
            agent.id,
            task.priority,
          ]
        );
        logger.info(
          { from: agent.name, to: action.to },
          '📋 Task delegated'
        );
      } else {
        logger.warn(
          { role: action.to },
          '⚠️ Delegation target role not found'
        );
      }
      break;
    }

    case 'message': {
      const recipientResult = await db.query(
        `SELECT id FROM agents
         WHERE company_id = $1
           AND role = $2
           AND status != 'terminated'
         LIMIT 1`,
        [agent.company_id, action.to]
      );
      const recipientId = recipientResult.rows[0]?.id || null;

      await db.query(
        `INSERT INTO messages (company_id, task_id, from_agent, to_agent, content, type)
         VALUES ($1, $2, $3, $4, $5, 'message')`,
        [agent.company_id, task.id, agent.id, recipientId, action.content]
      );
      logger.info(
        { from: agent.name, to: action.to },
        '💬 Message sent'
      );
      break;
    }

    case 'blocked':
      await releaseTask(task.id, 'blocked', action.reason);
      logger.warn(
        { agent: agent.name, reason: action.reason },
        '🚫 Task blocked'
      );
      break;

    case 'continue':
      // Save session state — agent will continue next heartbeat
      await db.query(
        `INSERT INTO agent_sessions (agent_id, task_id, state)
         VALUES ($1, $2, $3)
         ON CONFLICT (agent_id, task_id)
         DO UPDATE SET state = $3, updated_at = now()`,
        [agent.id, task.id, JSON.stringify({ progress: action.result })]
      );
      logger.info(
        { agent: agent.name, task: task.title },
        '⏳ Agent will continue next heartbeat'
      );
      break;
  }
}

/**
 * Log a heartbeat record.
 */
async function logHeartbeat(
  agentId: string,
  taskId: string | null,
  status: string,
  durationMs: number,
  costUsd: number,
  error?: any
) {
  await db.query(
    `INSERT INTO heartbeats (agent_id, task_id, status, duration_ms, cost_usd, details)
     VALUES ($1, $2, $3, $4, $5, $6)`,
    [
      agentId,
      taskId,
      status,
      durationMs,
      costUsd,
      error ? JSON.stringify({ error: String(error) }) : '{}',
    ]
  );
}

interface AgentRow {
  id: string;
  company_id: string;
  name: string;
  role: string;
  title: string;
  reports_to: string | null;
  runtime: string;
  model: string;
  system_prompt: string | null;
  status: string;
}

let intervalHandle: NodeJS.Timeout | null = null;

/**
 * Start the heartbeat scheduler.
 * Every HEARTBEAT_INTERVAL_MS, find all active agents and run their heartbeats.
 */
export function startScheduler() {
  logger.info(
    { intervalMs: env.HEARTBEAT_INTERVAL_MS },
    '⏰ Starting heartbeat scheduler'
  );

  // Immediate first run
  runHeartbeatCycle();

  intervalHandle = setInterval(
    runHeartbeatCycle,
    env.HEARTBEAT_INTERVAL_MS
  );
}

export function stopScheduler() {
  if (intervalHandle) {
    clearInterval(intervalHandle);
    intervalHandle = null;
    logger.info('⏰ Scheduler stopped');
  }
}

async function runHeartbeatCycle() {
  try {
    // Get all active (non-paused, non-terminated) agents
    const result = await db.query<AgentRow>(
      `SELECT * FROM agents
       WHERE status NOT IN ('paused', 'terminated')
       ORDER BY created_at ASC`
    );

    const agents = result.rows;

    if (agents.length === 0) {
      return; // No agents, quiet cycle
    }

    logger.debug(
      { count: agents.length },
      '💓 Heartbeat cycle — processing agents'
    );

    // Run heartbeats sequentially to avoid overwhelming APIs
    // (można zrobic Promise.allSettled dla paralelizmu)
    for (const agent of agents) {
      try {
        await agentHeartbeat(agent);
      } catch (err) {
        logger.error(
          { err, agent: agent.name },
          'Heartbeat failed for agent'
        );
      }
    }
  } catch (err) {
    logger.error({ err }, 'Heartbeat cycle failed');
  }
}

    subtasks: subtasks.rows,
  });
});

// Update task
router.patch('/tasks/:id', async (req, res) => {
  const { status, assigned_to, priority, description } = req.body;

  const sets: string[] = [];
  const params: any[] = [];

  if (status) {
    params.push(status);
    sets.push(`status = $${params.length}`);
  }
  if (assigned_to) {
    params.push(assigned_to);
    sets.push(`assigned_to = $${params.length}`);
  }
  if (priority !== undefined) {
    params.push(priority);
    sets.push(`priority = $${params.length}`);
  }
  if (description) {
    params.push(description);
    sets.push(`description = $${params.length}`);
  }

  if (sets.length === 0) return res.status(400).json({ error: 'Nothing to update' });

  params.push(req.params.id);
  const result = await db.query(
    `UPDATE tasks SET ${sets.join(', ')} WHERE id = $${params.length} RETURNING *`,
    params
  );

  res.json(result.rows[0]);
});

export default router;

router.post('/tasks/:taskId/messages', async (req, res) => {
  const parsed = createMessageSchema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error });

  const task = await db.query('SELECT company_id FROM tasks WHERE id = $1', [
    req.params.taskId,
  ]);
  if (task.rows.length === 0) return res.status(404).json({ error: 'Task not found' });

  const result = await db.query(
    `INSERT INTO messages (company_id, task_id, from_agent, to_agent, content, type)
     VALUES ($1, $2, NULL, $3, $4, $5)
     RETURNING *`,
    [
      task.rows[0].company_id,
      req.params.taskId,
      parsed.data.to_agent || null,
      parsed.data.content,
      parsed.data.type,
    ]
  );

  res.status(201).json(result.rows[0]);
});

import express from 'express';
import { env } from './env.js';
import { logger } from './utils/logger.js';
import { db } from './db/client.js';
import routes from './routes/index.js';
import { initRuntimes } from './runtimes/registry.js';
import { startScheduler, stopScheduler } from './orchestrator/scheduler.js';

const app = express();

app.use(express.json());

// Request logging
app.use((req, _res, next) => {
  logger.debug({ method: req.method, url: req.url }, 'Request');
  next();
});

// API routes
app.use('/api', routes);

// Error handler
app.use(
  (
    err: Error,
    _req: express.Request,
    res: express.Response,
    _next: express.NextFunction
  ) => {
    logger.error({ err }, 'Unhandled error');
    res.status(500).json({ error: 'Internal server error' });
  }
);

// Startup
async function start() {
  // Verify DB connection
  try {
    await db.query('SELECT 1');
    logger.info('✅ Database connected');
  } catch (err) {
    logger.fatal({ err }, '❌ Database connection failed');
    process.exit(1);
  }

  // Init runtimes
  initRuntimes();

  // Start HTTP server
  app.listen(env.PORT, () => {
    logger.info({ port: env.PORT }, '🚀 Autonomiczne Biuro is running');
  });

  // Start heartbeat scheduler
  startScheduler();
}

// Graceful shutdown
process.on('SIGINT', async () => {
  logger.info('Shutting down...');
  stopScheduler();
  await db.close();
  process.exit(0);
});

process.on('SIGTERM', async () => {
  logger.info('Shutting down...');
  stopScheduler();
  await db.close();
  process.exit(0);
});

start();
