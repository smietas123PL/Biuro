  config      JSONB DEFAULT '{}',
  schema      JSONB DEFAULT '{}',     -- JSON Schema parametrów wejściowych
  enabled     BOOLEAN DEFAULT true,
  created_at  TIMESTAMPTZ DEFAULT now(),
  updated_at  TIMESTAMPTZ DEFAULT now(),
  UNIQUE(company_id, name)
);

CREATE INDEX idx_tools_company ON tools(company_id);

  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  rate_limit  INT DEFAULT 100,        -- max calls per hour

);

import { db } from '../db/client.js';
import { logger } from '../utils/logger.js';

export interface ToolDefinition {
  id: string;
  name: string;
  type: string;
  description: string;
  config: any;
  schema: any;
  enabled: boolean;
}

export interface AgentToolPermission {
  tool: ToolDefinition;
  permissions: {
    read: boolean;
    write: boolean;
    execute: boolean;
  };
  rateLimit: number;
}

/**
 * Get all tools available to an agent, with permissions.
 */
export async function getAgentTools(
  agentId: string
): Promise<AgentToolPermission[]> {
  const result = await db.query(
    `
    SELECT
      t.*,
      at.permissions,
      at.rate_limit
    FROM agent_tools at
    JOIN tools t ON t.id = at.tool_id
    WHERE at.agent_id = $1
      AND t.enabled = true
    ORDER BY t.name ASC
    `,
    [agentId]
  );

  return result.rows.map((row) => ({
    tool: {
      id: row.id,
      name: row.name,
      type: row.type,
      description: row.description,
      config: row.config,
      schema: row.schema,
      enabled: row.enabled,
    },
    permissions: row.permissions,
    rateLimit: row.rate_limit,
  }));
}

/**
 * Check if agent can use a specific tool.
 */
export async function canUseTool(
  agentId: string,
  toolName: string,
  companyId: string
): Promise<{ allowed: boolean; tool?: ToolDefinition; reason?: string }> {
  // Find tool
  const toolResult = await db.query(
    'SELECT * FROM tools WHERE company_id = $1 AND name = $2',
    [companyId, toolName]
  );

  if (toolResult.rows.length === 0) {
    return { allowed: false, reason: `Tool "${toolName}" does not exist` };
  }

  const tool = toolResult.rows[0];

  if (!tool.enabled) {
    return { allowed: false, reason: `Tool "${toolName}" is disabled` };
  }

  // Check permission
  const permResult = await db.query(
    'SELECT * FROM agent_tools WHERE agent_id = $1 AND tool_id = $2',
    [agentId, tool.id]
  );

  if (permResult.rows.length === 0) {
    return {
      allowed: false,
      reason: `Agent does not have access to "${toolName}"`,
    };
  }

  const perm = permResult.rows[0];

  if (!perm.permissions.execute) {
    return {
      allowed: false,
      reason: `Agent lacks execute permission for "${toolName}"`,
    };
  }

  // Check rate limit
  const callCount = await db.query(
    `SELECT COUNT(*) as count FROM tool_calls
     WHERE agent_id = $1 AND tool_id = $2
       AND created_at > now() - interval '1 hour'`,
    [agentId, tool.id]
  );

  if (parseInt(callCount.rows[0].count) >= perm.rate_limit) {
    return {
      allowed: false,
      reason: `Rate limit exceeded for "${toolName}" (${perm.rate_limit}/hour)`,
    };
  }

  return { allowed: true, tool };
}

import { db } from '../db/client.js';
import { logger } from '../utils/logger.js';
import { canUseTool, type ToolDefinition } from './registry.js';
import { builtinTools } from './builtin/index.js';

export interface ToolCallResult {
  success: boolean;
  data?: any;
  error?: string;
  durationMs: number;
}

/**
 * Execute a tool call with full permission checking and logging.
 */
export async function executeTool(
  agentId: string,
  companyId: string,
  taskId: string | null,
  toolName: string,
  params: any
): Promise<ToolCallResult> {
  const startTime = Date.now();

  // 1. Permission check
  const { allowed, tool, reason } = await canUseTool(
    agentId,
    toolName,
    companyId
  );

  if (!allowed || !tool) {
    const result: ToolCallResult = {
      success: false,
      error: reason || 'Permission denied',
      durationMs: Date.now() - startTime,
    };

    await logToolCall(companyId, agentId, null, taskId, params, result, 'denied');
    return result;
  }

  // 2. Execute based on type
  try {
    let data: any;

    switch (tool.type) {
      case 'builtin':
        data = await executeBuiltin(tool, params);
        break;
      case 'http':
        data = await executeHttp(tool, params);
        break;
      case 'bash':
        data = await executeBash(tool, params);
        break;
      case 'function':
        data = await executeFunction(tool, params);
        break;
      default:
        throw new Error(`Unknown tool type: ${tool.type}`);
    }

    const result: ToolCallResult = {
      success: true,
      data,
      durationMs: Date.now() - startTime,
    };

    await logToolCall(companyId, agentId, tool.id, taskId, params, result, 'success');

    logger.info(
      { agentId, tool: toolName, durationMs: result.durationMs },
      '🔧 Tool executed'
    );

    return result;

  } catch (err: any) {
    const result: ToolCallResult = {
      success: false,
      error: err.message,
      durationMs: Date.now() - startTime,
    };

    await logToolCall(companyId, agentId, tool.id, taskId, params, result, 'error');

    logger.error({ err, tool: toolName }, '🔧 Tool execution failed');
    return result;
  }
}

// ---- Executors per type ----

async function executeBuiltin(tool: ToolDefinition, params: any): Promise<any> {
  const handler = builtinTools[tool.name];
  if (!handler) throw new Error(`Builtin tool "${tool.name}" not registered`);
  return handler(params, tool.config);
}

async function executeHttp(tool: ToolDefinition, params: any): Promise<any> {
  const { endpoint, method = 'POST', headers = {} } = tool.config;

  const url = endpoint.replace(
    /\{(\w+)\}/g,
    (_: string, key: string) => params[key] || ''
  );

  const res = await fetch(url, {
    method,
    headers: { 'Content-Type': 'application/json', ...headers },
    body: method !== 'GET' ? JSON.stringify(params) : undefined,
  });

  if (!res.ok) {
    throw new Error(`HTTP ${res.status}: ${await res.text()}`);
  }

  const contentType = res.headers.get('content-type') || '';
  if (contentType.includes('application/json')) {
    return res.json();
  }
  return { text: await res.text() };
}

async function executeBash(tool: ToolDefinition, params: any): Promise<any> {
  // SAFETY: Only execute whitelisted commands
  const { spawn } = await import('child_process');

  const allowedCommands = tool.config.allowed_commands || [];
  const command = params.command;

  if (!command) throw new Error('No command provided');

  const baseCmd = command.split(' ')[0];
  if (allowedCommands.length > 0 && !allowedCommands.includes(baseCmd)) {
    throw new Error(
      `Command "${baseCmd}" not in whitelist: ${allowedCommands.join(', ')}`
    );
  }

  return new Promise((resolve, reject) => {
    const timeout = tool.config.timeout_ms || 30000;
    let output = '';
    let errorOutput = '';

    const proc = spawn('sh', ['-c', command], {
      timeout,
      cwd: tool.config.working_dir || '/tmp',
      env: { ...process.env, ...tool.config.env },
    });

    proc.stdout.on('data', (data: Buffer) => {
      output += data.toString();
    });

    proc.stderr.on('data', (data: Buffer) => {
      errorOutput += data.toString();
    });

    proc.on('close', (code: number) => {
      if (code === 0) {
        resolve({ stdout: output.trim(), stderr: errorOutput.trim(), exitCode: code });
      } else {
        reject(new Error(`Exit code ${code}: ${errorOutput || output}`));
      }
    });

    proc.on('error', reject);
  });
}

async function executeFunction(tool: ToolDefinition, params: any): Promise<any> {
  // Custom function execution — extensible
  throw new Error(
    `Function tools require custom registration. Tool: ${tool.name}`
  );
}

// ---- Logging ----

async function logToolCall(
  companyId: string,
  agentId: string,
  toolId: string | null,
  taskId: string | null,
  params: any,
  result: ToolCallResult,
  status: string
) {
  if (toolId) {
    await db.query(
      `INSERT INTO tool_calls
        (company_id, agent_id, tool_id, task_id, params, result, status, duration_ms)
       VALUES ($1, $2, $3, $4, $5, $6, $7, $8)`,
      [
        companyId,
        agentId,
        toolId,
        taskId,
        JSON.stringify(params),
        JSON.stringify(result.data || { error: result.error }),
        status,
        result.durationMs,
      ]
    );
  }

  await db.query(
    `INSERT INTO audit_log
      (company_id, agent_id, action, entity_type, entity_id, details)
     VALUES ($1, $2, $3, 'tool', $4, $5)`,
    [
      companyId,
      agentId,
      `tool.${status}`,
      toolId,
      JSON.stringify({
        tool: toolId,
        params,
        result: result.success ? 'ok' : result.error,
        duration_ms: result.durationMs,
      }),
    ]
  );
}

import fs from 'fs/promises';
import path from 'path';

const WORKSPACE_ROOT = process.env.WORKSPACE_ROOT || '/tmp/biuro-workspace';

function sanitizePath(filePath: string): string {
  // Prevent directory traversal
  const resolved = path.resolve(WORKSPACE_ROOT, filePath);
  if (!resolved.startsWith(WORKSPACE_ROOT)) {
    throw new Error('Path traversal detected');
  }
  return resolved;
}

export async function fileWrite(
  params: { path: string; content: string },
  _config: any
): Promise<any> {
  const fullPath = sanitizePath(params.path);
  await fs.mkdir(path.dirname(fullPath), { recursive: true });
  await fs.writeFile(fullPath, params.content, 'utf-8');
  return { written: fullPath, bytes: Buffer.byteLength(params.content) };
}

export async function fileRead(
  params: { path: string },
  _config: any
): Promise<any> {
  const fullPath = sanitizePath(params.path);
  const content = await fs.readFile(fullPath, 'utf-8');
  return { path: fullPath, content };
}

export async function httpRequest(
  params: {
    url: string;
    method?: string;
    headers?: Record<string, string>;
    body?: any;
  },
  config: { allowed_domains?: string[] }
): Promise<any> {
  // Domain whitelist
  if (config.allowed_domains && config.allowed_domains.length > 0) {
    const url = new URL(params.url);
    if (!config.allowed_domains.includes(url.hostname)) {
      throw new Error(
        `Domain "${url.hostname}" not in whitelist: ${config.allowed_domains.join(', ')}`
      );
    }
  }

  const res = await fetch(params.url, {
    method: params.method || 'GET',
    headers: params.headers || {},
    body: params.body ? JSON.stringify(params.body) : undefined,
  });

  const contentType = res.headers.get('content-type') || '';
  const data = contentType.includes('json')
    ? await res.json()
    : await res.text();

  return {
    status: res.status,
    headers: Object.fromEntries(res.headers.entries()),
    data,
  };
}

import { db } from '../db/client.js';
import { logger } from '../utils/logger.js';

export interface Policy {
  id: string;
  company_id: string;
  name: string;
  type: string;
  rules: any;
  enabled: boolean;
}

/**
 * Check all policies before an agent action.
 * Returns { allowed: true } or { allowed: false, reason, requiresApproval, policyId }
 */
export async function checkPolicies(
  companyId: string,
  agentId: string,
  action: string,
  context: any
): Promise<{
  allowed: boolean;
  reason?: string;
  requiresApproval?: boolean;
  policyId?: string;
}> {
  const result = await db.query<Policy>(
    'SELECT * FROM policies WHERE company_id = $1 AND enabled = true',
    [companyId]
  );

  for (const policy of result.rows) {
    const check = await evaluatePolicy(policy, agentId, action, context);
    if (!check.allowed) {
      logger.info(
        { policy: policy.name, agent: agentId, action },
        '🛡️ Policy triggered'
      );
      return {
        allowed: false,
        reason: check.reason,
        requiresApproval: check.requiresApproval,
        policyId: policy.id,
      };
    }
  }

  return { allowed: true };
}

async function evaluatePolicy(
  policy: Policy,
  agentId: string,
  action: string,
  context: any
): Promise<{
  allowed: boolean;
  reason?: string;
  requiresApproval?: boolean;
}> {
  switch (policy.type) {
    case 'approval_required':
      return evaluateApprovalPolicy(policy, agentId, action);

    case 'budget_threshold':
      return evaluateBudgetPolicy(policy, agentId, context);

    case 'delegation_limit':
      return evaluateDelegationPolicy(policy, context);

    case 'rate_limit':
      return evaluateRatePolicy(policy, agentId, action);

    case 'tool_restriction':
      return evaluateToolPolicy(policy, agentId, action, context);

    default:
      return { allowed: true };
  }
}

// ---- Policy evaluators ----

async function evaluateApprovalPolicy(
  policy: Policy,
  agentId: string,
  action: string
): Promise<{ allowed: boolean; reason?: string; requiresApproval?: boolean }> {
  const { actions = [], roles = [] } = policy.rules;

  // Check if this action type requires approval
  if (actions.length > 0 && !actions.includes(action)) {
    return { allowed: true };
  }

  // Check if this agent's role requires approval
  if (roles.length > 0) {
    const agent = await db.query(
      'SELECT role FROM agents WHERE id = $1',
      [agentId]
    );
    if (agent.rows.length > 0 && !roles.includes(agent.rows[0].role)) {
      return { allowed: true };
    }
  }

  // Check if there's already an approved approval for this
  const existing = await db.query(
    `SELECT * FROM approvals
     WHERE requested_by = $1
       AND type = $2
       AND status = 'approved'
       AND created_at > now() - interval '24 hours'
     LIMIT 1`,
    [agentId, action]
  );

  if (existing.rows.length > 0) {
    return { allowed: true }; // Already approved
  }

  return {
    allowed: false,
    reason: `Policy "${policy.name}" requires approval for action "${action}"`,
    requiresApproval: true,
  };
}

async function evaluateBudgetPolicy(
  policy: Policy,
  agentId: string,
  context: any
): Promise<{ allowed: boolean; reason?: string; requiresApproval?: boolean }> {
  const { threshold_pct = 80 } = policy.rules;

  const budget = await db.query(
    `SELECT * FROM budgets
     WHERE agent_id = $1
       AND month = date_trunc('month', now())::date`,
    [agentId]
  );

  if (budget.rows.length === 0) return { allowed: true };

  const b = budget.rows[0];
  const usedPct = (parseFloat(b.spent_usd) / parseFloat(b.limit_usd)) * 100;

  if (usedPct >= threshold_pct) {
    return {
      allowed: false,
      reason: `Budget ${usedPct.toFixed(1)}% used (threshold: ${threshold_pct}%)`,
      requiresApproval: true,
    };
  }

  return { allowed: true };
}

async function evaluateDelegationPolicy(
  policy: Policy,
  context: any
): Promise<{ allowed: boolean; reason?: string }> {
  const { max_depth = 5 } = policy.rules;

  if (context.delegationDepth && context.delegationDepth >= max_depth) {
    return {
      allowed: false,
      reason: `Delegation depth ${context.delegationDepth} exceeds max ${max_depth}`,
    };
  }

  return { allowed: true };
}

async function evaluateRatePolicy(
  policy: Policy,
  agentId: string,
  action: string
): Promise<{ allowed: boolean; reason?: string }> {
  const { max_per_hour = 60, actions = [] } = policy.rules;

  if (actions.length > 0 && !actions.includes(action)) {
    return { allowed: true };
  }

  const count = await db.query(
    `SELECT COUNT(*) as count FROM audit_log
     WHERE agent_id = $1
       AND action = $2
       AND created_at > now() - interval '1 hour'`,
    [agentId, action]
  );

  if (parseInt(count.rows[0].count) >= max_per_hour) {
    return {
      allowed: false,
      reason: `Rate limit: ${max_per_hour}/hour exceeded for "${action}"`,
    };
  }

  return { allowed: true };
}

async function evaluateToolPolicy(
  policy: Policy,
  agentId: string,
  action: string,
  context: any
): Promise<{ allowed: boolean; reason?: string }> {
  if (action !== 'use_tool') return { allowed: true };

  const { blocked_tools = [], roles = [] } = policy.rules;

  // Check role
  if (roles.length > 0) {
    const agent = await db.query('SELECT role FROM agents WHERE id = $1', [agentId]);
    if (!roles.includes(agent.rows[0]?.role)) {
      return { allowed: true };
    }
  }

  // Check tool name
  if (blocked_tools.includes(context.toolName)) {
    return {
      allowed: false,
      reason: `Tool "${context.toolName}" is blocked by policy "${policy.name}"`,
    };
  }

  return { allowed: true };
}

export interface SafetyLimits {
  maxHeartbeatsPerHour: number;
  maxToolCallsPerTask: number;
  maxDelegationDepth: number;
  maxMessagesPerMinute: number;
  maxConsecutiveErrors: number;
  maxTaskDurationHours: number;
}

const DEFAULT_LIMITS: SafetyLimits = {
  maxHeartbeatsPerHour: 60,
  maxToolCallsPerTask: 100,
  maxDelegationDepth: 5,
  maxMessagesPerMinute: 20,
  maxConsecutiveErrors: 5,
  maxTaskDurationHours: 24,
};

/**
 * Run all safety checks before an agent heartbeat.
 * Returns false if the agent should NOT proceed.
 */
export async function checkSafety(
  agentId: string,
  companyId: string,
  taskId: string | null,
  customLimits?: Partial<SafetyLimits>
): Promise<{ safe: boolean; reason?: string }> {
  const limits = { ...DEFAULT_LIMITS, ...customLimits };

  // 1. Heartbeat frequency
  const heartbeatCount = await db.query(
    `SELECT COUNT(*) as count FROM heartbeats
     WHERE agent_id = $1 AND created_at > now() - interval '1 hour'`,
    [agentId]
  );

  if (parseInt(heartbeatCount.rows[0].count) >= limits.maxHeartbeatsPerHour) {
    await autoPause(agentId, companyId, 'Too many heartbeats per hour — possible loop');
    return { safe: false, reason: 'heartbeat_rate_exceeded' };
  }

  // 2. Tool calls on current task
  if (taskId) {
    const toolCallCount = await db.query(
      `SELECT COUNT(*) as count FROM tool_calls
       WHERE task_id = $1`,
      [taskId]
    );

    if (parseInt(toolCallCount.rows[0].count) >= limits.maxToolCallsPerTask) {
      await blockTask(taskId, 'Too many tool calls — possible loop');
      return { safe: false, reason: 'tool_call_limit' };
    }
  }

  // 3. Delegation depth
  if (taskId) {
    const depth = await getDelegationDepth(taskId);
    if (depth >= limits.maxDelegationDepth) {
      await blockTask(taskId, `Delegation depth ${depth} exceeds max ${limits.maxDelegationDepth}`);
      return { safe: false, reason: 'delegation_depth_exceeded' };
    }
  }

  // 4. Message flood
  const msgCount = await db.query(
    `SELECT COUNT(*) as count FROM messages
     WHERE from_agent = $1 AND created_at > now() - interval '1 minute'`,
    [agentId]
  );

  if (parseInt(msgCount.rows[0].count) >= limits.maxMessagesPerMinute) {
    await autoPause(agentId, companyId, 'Message flood detected');
    return { safe: false, reason: 'message_flood' };
  }

  // 5. Consecutive errors
  const recentHeartbeats = await db.query(
    `SELECT status FROM heartbeats
     WHERE agent_id = $1
     ORDER BY created_at DESC
     LIMIT $2`,
    [agentId, limits.maxConsecutiveErrors]
  );

  const consecutiveErrors = recentHeartbeats.rows
    .map((r) => r.status)
    .filter((s) => s === 'error').length;

  if (consecutiveErrors >= limits.maxConsecutiveErrors) {
    await autoPause(
      agentId,
      companyId,
      `${consecutiveErrors} consecutive errors`
    );
    return { safe: false, reason: 'consecutive_errors' };
  }

  // 6. Task stuck too long
  if (taskId) {
    const taskAge = await db.query(
      `SELECT EXTRACT(EPOCH FROM (now() - locked_at)) / 3600 as hours
       FROM tasks WHERE id = $1 AND locked_at IS NOT NULL`,
      [taskId]
    );

    if (
      taskAge.rows.length > 0 &&
      parseFloat(taskAge.rows[0].hours) >= limits.maxTaskDurationHours
    ) {
      await blockTask(
        taskId,
        `Task has been in progress for ${parseFloat(taskAge.rows[0].hours).toFixed(1)} hours`
      );
      return { safe: false, reason: 'task_timeout' };
    }
  }

  // 7. Circular delegation detection
  if (taskId) {
    const isCircular = await detectCircularDelegation(taskId, agentId);
    if (isCircular) {
      await blockTask(taskId, 'Circular delegation detected');
      await autoPause(agentId, companyId, 'Circular delegation detected');
      return { safe: false, reason: 'circular_delegation' };
    }
  }

  return { safe: true };
}

/**
 * Calculate delegation depth by walking up parent_id chain.
 */
async function getDelegationDepth(taskId: string): Promise<number> {
  let depth = 0;
  let currentId: string | null = taskId;

  while (currentId) {
    const result = await db.query(
      'SELECT parent_id FROM tasks WHERE id = $1',
      [currentId]
    );
    if (result.rows.length === 0 || !result.rows[0].parent_id) break;
    currentId = result.rows[0].parent_id;
    depth++;
    if (depth > 20) break; // Safety cap
  }

  return depth;
}

/**
 * Detect circular delegation: agent A → B → C → A
 */
async function detectCircularDelegation(
  taskId: string,
  agentId: string
): Promise<boolean> {
  const visited = new Set<string>();
  let currentId: string | null = taskId;

  while (currentId) {
    if (visited.has(currentId)) return true;
    visited.add(currentId);

    const result = await db.query(
      'SELECT parent_id, assigned_to FROM tasks WHERE id = $1',
      [currentId]
    );

    if (result.rows.length === 0) break;

    // Check if a parent task is assigned back to the same agent
    if (result.rows[0].parent_id) {
      const parentTask = await db.query(
        'SELECT assigned_to FROM tasks WHERE id = $1',
        [result.rows[0].parent_id]
      );
      if (parentTask.rows[0]?.assigned_to === agentId && visited.size > 1) {
        return true;
      }
    }

    currentId = result.rows[0].parent_id;
    if (visited.size > 20) break;
  }

  return false;
}

async function autoPause(

     WHERE id = $1`,
    [taskId, JSON.stringify(reason)]
  );

  logger.warn({ taskId, reason }, '🚫 Task blocked by safety system');
}

import { WebSocketServer, WebSocket } from 'ws';
import type { Server } from 'http';
import { logger } from '../utils/logger.js';

interface Client {
  ws: WebSocket;
  companyId: string;
  connectedAt: Date;
}

const clients: Set<Client> = new Set();

let wss: WebSocketServer;

/**
 * Initialize WebSocket server on the existing HTTP server.
 */
export function initWebSocket(server: Server): void {
  wss = new WebSocketServer({ server, path: '/ws' });

  wss.on('connection', (ws, req) => {
    // Extract company ID from query params: /ws?company=<uuid>
    const url = new URL(req.url || '', `http://${req.headers.host}`);
    const companyId = url.searchParams.get('company') || '';

    if (!companyId) {
      ws.close(4001, 'Missing company parameter');
      return;
    }

    const client: Client = {
      ws,
      companyId,
      connectedAt: new Date(),
    };

    clients.add(client);
    logger.info(
      { companyId, totalClients: clients.size },
      '🔌 WebSocket client connected'
    );

    // Send welcome
    ws.send(
      JSON.stringify({
        type: 'connected',
        data: { companyId, timestamp: new Date().toISOString() },
      })
    );

    ws.on('close', () => {
      clients.delete(client);
      logger.debug(
        { companyId, totalClients: clients.size },
        '🔌 WebSocket client disconnected'
      );
    });

    ws.on('error', (err) => {
      logger.error({ err }, 'WebSocket client error');
      clients.delete(client);
    });

    // Handle incoming messages (future: commands from dashboard)
    ws.on('message', (data) => {
      try {
        const msg = JSON.parse(data.toString());
        handleClientMessage(client, msg);
      } catch {
        // ignore malformed
      }
    });
  });

  logger.info('🔌 WebSocket server initialized on /ws');
}

/**
 * Broadcast a message to all clients watching a specific company.
 */
export function broadcast(companyId: string, message: any): void {
  const payload = JSON.stringify({
    ...message,
    timestamp: new Date().toISOString(),
  });

  let sent = 0;
  for (const client of clients) {
    if (client.companyId === companyId && client.ws.readyState === WebSocket.OPEN) {
      client.ws.send(payload);
      sent++;
    }
  }

  if (sent > 0) {
    logger.debug({ companyId, sent, type: message.type }, '📡 Broadcast');
  }
}

/**
 * Broadcast to ALL companies (system events).
 */
export function broadcastAll(message: any): void {
  const payload = JSON.stringify({
    ...message,
    timestamp: new Date().toISOString(),
  });

  for (const client of clients) {
    if (client.ws.readyState === WebSocket.OPEN) {
      client.ws.send(payload);
    }
  }
}

/**
 * Handle messages from dashboard clients.
 */
function handleClientMessage(client: Client, msg: any): void {
  switch (msg.type) {
    case 'ping':
      client.ws.send(JSON.stringify({ type: 'pong' }));
      break;

    case 'subscribe':
      // Client can switch company
      if (msg.companyId) {
        client.companyId = msg.companyId;
      }
      break;

    default:
      break;
  }
}

/**
 * Get connection stats.
 */
export function getWsStats() {
  const byCompany: Record<string, number> = {};
  for (const client of clients) {
    byCompany[client.companyId] = (byCompany[client.companyId] || 0) + 1;
  }
  return {
    totalConnections: clients.size,
    byCompany,
  };
}

     WHERE company_id = $1
     ORDER BY created_at ASC`,

  options?: RequestInit
): Promise<T> {

}

import { useEffect, useRef, useState, useCallback } from 'react';

export interface WsMessage {
  type: string;
  data: any;
  timestamp: string;
}

export function useWebSocket(companyId: string | null) {
  const wsRef = useRef<WebSocket | null>(null);
  const [connected, setConnected] = useState(false);
  const [messages, setMessages] = useState<WsMessage[]>([]);
  const [lastMessage, setLastMessage] = useState<WsMessage | null>(null);

  useEffect(() => {
    if (!companyId) return;

    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
    const wsUrl = `${protocol}//${window.location.host}/ws?company=${companyId}`;

    const ws = new WebSocket(wsUrl);
    wsRef.current = ws;

    ws.onopen = () => setConnected(true);

    ws.onmessage = (event) => {
      try {
        const msg: WsMessage = JSON.parse(event.data);
        setLastMessage(msg);
        setMessages((prev) => [msg, ...prev].slice(0, 100)); // keep last 100
      } catch {
        // ignore
      }
    };

    ws.onclose = () => {
      setConnected(false);
      // Auto-reconnect after 3s
      setTimeout(() => {
        if (wsRef.current === ws) {
          // Trigger re-render to reconnect
          setConnected(false);
        }
      }, 3000);
    };

    ws.onerror = () => setConnected(false);

    return () => {
      ws.close();
    };
  }, [companyId]);

  const clearMessages = useCallback(() => setMessages([]), []);

  return { connected, messages, lastMessage, clearMessages };
}

import { useState, useEffect, useCallback } from 'react';

export function useApi<T>(
  fetcher: () => Promise<T>,
  deps: any[] = []
) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refetch = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await fetcher();
      setData(result);
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, deps);

  useEffect(() => {
    refetch();
  }, [refetch]);

  return { data, loading, error, refetch };
}

import { BudgetsPage } from './pages/BudgetsPage';
import { api } from './api/client';

export default function App() {
  const [companyId, setCompanyId] = useState<string | null>(null);
  const [companies, setCompanies] = useState<any[]>([]);

  useEffect(() => {
    api.listCompanies().then((list) => {
      setCompanies(list);
      if (list.length > 0) {
        setCompanyId(list[0].id);
      }
    });
  }, []);

  if (!companyId) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center">
          <h1 className="text-3xl font-bold mb-4">🏢 Autonomiczne Biuro</h1>
          {companies.length === 0 ? (
            <div>
              <p className="text-gray-500 mb-4">
                No companies found. Create one via API:
              </p>
              <pre className="bg-gray-900 rounded-lg p-4 text-sm text-left">
                {`curl -X POST http://localhost:3100/api/companies \\
  -H "Content-Type: application/json" \\
  -d '{"name": "My AI Company", "mission": "..."}'`}
              </pre>
            </div>
          ) : (
            <div className="space-y-2">
              <p className="text-gray-500">Select a company:</p>
              {companies.map((c) => (
                <button
                  key={c.id}
                  onClick={() => setCompanyId(c.id)}
                  className="block w-full bg-gray-900 hover:bg-gray-800 border border-gray-700 rounded-lg p-4 text-left"
                >
                  <span className="font-medium">{c.name}</span>
                  {c.mission && (
                    <p className="text-sm text-gray-500 mt-1">
                      {c.mission}
                    </p>
                  )}
                </button>
              ))}
            </div>
          )}
        </div>
      </div>
    );
  }

  return (
    <Layout>
      {/* Company selector */}
      {companies.length > 1 && (
        <div className="mb-4">
          <select
            value={companyId}
            onChange={(e) => setCompanyId(e.target.value)}
            className="bg-gray-800 border border-gray-700 rounded px-3 py-1 text-sm"
          >
            {companies.map((c) => (
              <option key={c.id} value={c.id}>
                {c.name}
              </option>
            ))}
          </select>
        </div>
      )}

      <Routes>
        <Route
          path="/"
          element={<DashboardPage companyId={companyId} />}
        />
        <Route
          path="/agents"
          element={<AgentsPage companyId={companyId} />}
        />
        <Route
          path="/tasks"
          element={<TasksPage companyId={companyId} />}
        />
        <Route
          path="/org-chart"
          element={<OrgChartPage companyId={companyId} />}
        />
        <Route
          path="/approvals"
          element={<ApprovalsPage companyId={companyId} />}
        />
        <Route
          path="/budgets"
          element={<BudgetsPage companyId={companyId} />}
        />
        <Route
          path="/audit-log"
          element={<AuditLogPage companyId={companyId} />}
        />
      </Routes>
    </Layout>
  );
}
