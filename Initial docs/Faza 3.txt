import React, { useEffect, useRef } from 'react';
import { X } from 'lucide-react';

interface ModalProps {
  open: boolean;
  onClose: () => void;
  title: string;
  children: React.ReactNode;
  wide?: boolean;
}

export function Modal({ open, onClose, title, children, wide }: ModalProps) {
  const overlayRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const handleEsc = (e: KeyboardEvent) => {
      if (e.key === 'Escape') onClose();
    };
    if (open) document.addEventListener('keydown', handleEsc);
    return () => document.removeEventListener('keydown', handleEsc);
  }, [open, onClose]);

  if (!open) return null;

  return (
    <div
      ref={overlayRef}
      className="fixed inset-0 z-50 flex items-center justify-center bg-black/60 backdrop-blur-sm p-4"
      onClick={(e) => e.target === overlayRef.current && onClose()}
    >
      <div
        className={`bg-gray-900 border border-gray-700 rounded-xl shadow-2xl w-full ${
          wide ? 'max-w-2xl' : 'max-w-lg'
        } max-h-[90vh] flex flex-col`}
      >
        <div className="flex items-center justify-between px-6 py-4 border-b border-gray-800">
          <h2 className="text-lg font-semibold">{title}</h2>
          <button
            onClick={onClose}
            className="text-gray-500 hover:text-gray-300 transition-colors"
          >
            <X size={20} />
          </button>
        </div>
        <div className="overflow-y-auto px-6 py-4 flex-1">{children}</div>
      </div>
    </div>
  );
}

import React from 'react';

interface InputProps extends React.InputHTMLAttributes<HTMLInputElement> {
  label: string;
  error?: string;
}

export function Input({ label, error, className, ...props }: InputProps) {
  return (
    <div className="space-y-1">
      <label className="text-sm text-gray-400">{label}</label>
      <input
        className={`w-full bg-gray-800 border rounded-lg px-3 py-2 text-sm
          focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent
          ${error ? 'border-red-500' : 'border-gray-700'}
          ${className || ''}`}
        {...props}
      />
      {error && <p className="text-xs text-red-400">{error}</p>}
    </div>
  );

interface SelectProps extends React.SelectHTMLAttributes<HTMLSelectElement> {
  label: string;
  options: { value: string; label: string }[];
  error?: string;
}

export function Select({
  label,
  options,
  error,
  className,
  ...props
}: SelectProps) {
  return (
    <div className="space-y-1">
      <label className="text-sm text-gray-400">{label}</label>
      <select
        className={`w-full bg-gray-800 border rounded-lg px-3 py-2 text-sm
          focus:outline-none focus:ring-2 focus:ring-blue-500
          ${error ? 'border-red-500' : 'border-gray-700'}
          ${className || ''}`}
        {...props}
      >
        {options.map((opt) => (
          <option key={opt.value} value={opt.value}>
            {opt.label}
          </option>
        ))}
      </select>
      {error && <p className="text-xs text-red-400">{error}</p>}
    </div>
  );
}

export function Button({
  variant = 'primary',
  loading,
  children,
  className,
  ...props
}: React.ButtonHTMLAttributes<HTMLButtonElement> & {
  variant?: 'primary' | 'secondary' | 'danger' | 'ghost';
  loading?: boolean;
}) {
  const variants = {
    primary: 'bg-blue-600 hover:bg-blue-700 text-white',
    secondary: 'bg-gray-700 hover:bg-gray-600 text-gray-200',
    danger: 'bg-red-700 hover:bg-red-600 text-white',
    ghost: 'bg-transparent hover:bg-gray-800 text-gray-400',
  };

  return (
    <button
      className={`px-4 py-2 rounded-lg text-sm font-medium transition-colors
        disabled:opacity-50 disabled:cursor-not-allowed
        ${variants[variant]} ${className || ''}`}
      disabled={loading || props.disabled}
      {...props}
    >
      {loading ? (
        <span className="flex items-center gap-2">
          <span className="animate-spin">⏳</span> Loading...
        </span>
      ) : (
        children
      )}
    </button>
  );
}

import React from 'react';

const ROLE_COLORS: Record<string, string> = {
  ceo: 'bg-purple-600',
  cto: 'bg-blue-600',
  developer: 'bg-green-600',
  designer: 'bg-pink-600',
  marketer: 'bg-orange-600',
  pm: 'bg-teal-600',
  qa: 'bg-yellow-600',
  support: 'bg-cyan-600',
};

export function AgentAvatar({
  name,
  role,
  size = 'md',
}: {
  name: string;
  role: string;
  size?: 'sm' | 'md' | 'lg';
}) {
  const initials = name
    .split(' ')
    .map((w) => w[0])
    .join('')
    .toUpperCase()
    .slice(0, 2);

  const color = ROLE_COLORS[role] || 'bg-gray-600';
  const sizeClasses = {
    sm: 'w-7 h-7 text-xs',
    md: 'w-9 h-9 text-sm',
    lg: 'w-12 h-12 text-base',
  };

  return (
    <div
      className={`${color} ${sizeClasses[size]} rounded-full flex items-center justify-center font-bold text-white`}
      title={`${name} (${role})`}
    >
      {initials}
    </div>
  );
}

React
import React, { useState } from 'react';
import { AgentAvatar } from './AgentAvatar';
import { Button } from './Form';
import { api } from '../api/client';
import { Send } from 'lucide-react';

interface Message {
  id: string;
  from_agent: string | null;
  from_name: string | null;
  content: string;
  type: string;
  metadata: any;
  created_at: string;
}

export function MessageThread({
  messages,
  taskId,
  onMessageSent,
}: {
  messages: Message[];
  taskId: string;
  onMessageSent?: () => void;
}) {
  const [input, setInput] = useState('');
  const [sending, setSending] = useState(false);

  const handleSend = async () => {
    if (!input.trim()) return;
    setSending(true);
    try {
      await api.sendMessage(taskId, input.trim());
      setInput('');
      onMessageSent?.();
    } finally {
      setSending(false);
    }
  };

  const typeStyles: Record<string, string> = {
    message: 'border-gray-800',
    heartbeat_log: 'border-blue-900/50 bg-blue-950/20',
    tool_call: 'border-yellow-900/50 bg-yellow-950/20',
    tool_result: 'border-green-900/50 bg-green-950/20',
    delegation: 'border-purple-900/50 bg-purple-950/20',
    status_update: 'border-gray-800 bg-gray-900/50',
  };

  const typeIcons: Record<string, string> = {
    message: '💬',
    heartbeat_log: '💓',
    tool_call: '🔧',
    tool_result: '📦',
    delegation: '📋',
    status_update: '📊',
    approval_request: '🎫',
  };

  return (
    <div className="flex flex-col h-full">
      {/* Messages */}
      <div className="flex-1 overflow-y-auto space-y-3 mb-4">
        {messages.length === 0 ? (
          <p className="text-gray-600 text-sm text-center py-8">
            No messages yet
          </p>
        ) : (
          messages.map((msg) => {
            const isBoard = !msg.from_agent;
            const isSystem = ['heartbeat_log', 'tool_call', 'tool_result'].includes(msg.type);

            return (
              <div
                key={msg.id}
                className={`rounded-lg border p-3 ${typeStyles[msg.type] || 'border-gray-800'}`}
              >
                <div className="flex items-start gap-3">
                  {!isBoard && !isSystem && (
                    <AgentAvatar
                      name={msg.from_name || 'Agent'}
                      role="agent"
                      size="sm"
                    />
                  )}
                  <div className="flex-1 min-w-0">
                    <div className="flex items-center gap-2 mb-1">
                      <span className="text-xs">
                        {typeIcons[msg.type] || '📝'}
                      </span>
                      <span className="text-sm font-medium text-gray-300">
                        {isBoard ? '🧑 Board' : msg.from_name || 'System'}
                      </span>
                      <span className="text-xs text-gray-600">
                        {msg.type}
                      </span>
                      <span className="text-xs text-gray-700 ml-auto">
                        {new Date(msg.created_at).toLocaleTimeString()}
                      </span>
                    </div>

                    {msg.type === 'heartbeat_log' ? (
                      <details className="text-sm">
                        <summary className="text-gray-500 cursor-pointer hover:text-gray-300">
                          Agent response (click to expand)
                        </summary>
                        <pre className="mt-2 text-xs text-gray-400 overflow-x-auto whitespace-pre-wrap bg-gray-900 rounded p-2">
                          {msg.content}
                        </pre>
                      </details>
                    ) : (
                      <p className="text-sm text-gray-300 whitespace-pre-wrap">
                        {msg.content}
                      </p>
                    )}

                    {msg.metadata &&
                      msg.type === 'tool_result' &&
                      msg.metadata.data && (
                        <pre className="mt-2 text-xs text-gray-500 overflow-x-auto bg-gray-900 rounded p-2">
                          {JSON.stringify(msg.metadata.data, null, 2).slice(0, 500)}
                        </pre>
                      )}
                  </div>
                </div>
              </div>
            );
          })
        )}
      </div>

      {/* Input */}
      <div className="flex gap-2 border-t border-gray-800 pt-3">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && !e.shiftKey && handleSend()}
          placeholder="Send a message as Board..."
          className="flex-1 bg-gray-800 border border-gray-700 rounded-lg px-3 py-2 text-sm
            focus:outline-none focus:ring-2 focus:ring-blue-500"
        />
        <Button onClick={handleSend} loading={sending} variant="primary">
          <Send size={16} />
        </Button>
      </div>
    </div>
  );
}

import React, { useState } from 'react';
import { StatusBadge } from './StatusBadge';
import { ChevronRight, ChevronDown, Target, Plus } from 'lucide-react';

interface Goal {
  id: string;
  title: string;
  description: string;
  status: string;
  children: Goal[];
}

function GoalNode({
  goal,
  depth = 0,
  onAddChild,
}: {
  goal: Goal;
  depth?: number;
  onAddChild: (parentId: string) => void;
}) {
  const [expanded, setExpanded] = useState(true);
  const hasChildren = goal.children.length > 0;

  return (
    <div className={depth > 0 ? 'ml-6' : ''}>
      <div className="flex items-start gap-2 py-2 group">
        <button
          onClick={() => setExpanded(!expanded)}
          className="mt-0.5 text-gray-600 hover:text-gray-400"
        >
          {hasChildren ? (
            expanded ? (
              <ChevronDown size={16} />
            ) : (
              <ChevronRight size={16} />
            )
          ) : (
            <Target size={16} className="text-gray-700" />
          )}
        </button>

        <div className="flex-1">
          <div className="flex items-center gap-2">
            <span className="font-medium text-sm">{goal.title}</span>
            <StatusBadge status={goal.status} />
          </div>
          {goal.description && (
            <p className="text-xs text-gray-500 mt-0.5">
              {goal.description}
            </p>
          )}
        </div>

        <button
          onClick={() => onAddChild(goal.id)}
          className="opacity-0 group-hover:opacity-100 text-gray-600 hover:text-blue-400 transition-opacity"
          title="Add sub-goal"
        >
          <Plus size={14} />
        </button>
      </div>

      {expanded &&
        goal.children.map((child) => (
          <GoalNode
            key={child.id}
            goal={child}
            depth={depth + 1}
            onAddChild={onAddChild}
          />
        ))}
    </div>
  );
}

export function GoalTree({
  goals,
  onAddGoal,
}: {
  goals: Goal[];
  onAddGoal: (parentId?: string) => void;
}) {
  return (
    <div className="bg-gray-900 rounded-lg border border-gray-800 p-4">
      {goals.length === 0 ? (
        <p className="text-gray-600 text-center py-4">No goals defined</p>
      ) : (
        goals.map((goal) => (
          <GoalNode
            key={goal.id}
            goal={goal}
            onAddChild={(parentId) => onAddGoal(parentId)}
          />
        ))
      )}
    </div>
  );
}

import React, { useState } from 'react';

  open,

  onCreated,

  );
}

import { Router } from 'express';
import { db } from '../db/client.js';

const router = Router();

// Get budget for agent
router.get('/agents/:agentId/budgets', async (req, res) => {
  const result = await db.query(
    `SELECT * FROM budgets
     WHERE agent_id = $1
     ORDER BY month DESC`,
    [req.params.agentId]
  );
  res.json(result.rows);
});

// Set budget for agent
router.post('/agents/:agentId/budgets', async (req, res) => {
  const { month, limit_usd } = req.body;
  const budgetMonth =
    month || new Date().toISOString().slice(0, 7) + '-01';

  const result = await db.query(
    `INSERT INTO budgets (agent_id, month, limit_usd)
     VALUES ($1, $2, $3)
     ON CONFLICT (agent_id, month)
     DO UPDATE SET limit_usd = $3
     RETURNING *`,
    [req.params.agentId, budgetMonth, limit_usd]
  );
  res.status(201).json(result.rows[0]);
});

// Company-wide cost summary
router.get('/companies/:companyId/costs', async (req, res) => {
  const { period } = req.query; // 'day', 'week', 'month'

  let interval = '30 days';
  if (period === 'day') interval = '1 day';
  if (period === 'week') interval = '7 days';

  const [byAgent, byDay, total] = await Promise.all([
    db.query(
      `SELECT a.name, a.role,
         COALESCE(SUM(al.cost_usd), 0) as total_cost,
         COUNT(*) as action_count
       FROM agents a
       LEFT JOIN audit_log al ON al.agent_id = a.id
         AND al.created_at > now() - interval '${interval}'
         AND al.cost_usd > 0
       WHERE a.company_id = $1
       GROUP BY a.id, a.name, a.role
       ORDER BY total_cost DESC`,
      [req.params.companyId]
    ),
    db.query(
      `SELECT DATE(al.created_at) as day,
         COALESCE(SUM(al.cost_usd), 0) as total_cost
       FROM audit_log al
       WHERE al.company_id = $1
         AND al.created_at > now() - interval '${interval}'
         AND al.cost_usd > 0
       GROUP BY DATE(al.created_at)
       ORDER BY day ASC`,
      [req.params.companyId]
    ),
    db.query(
      `SELECT COALESCE(SUM(cost_usd), 0) as total
       FROM audit_log
       WHERE company_id = $1
         AND created_at > now() - interval '${interval}'`,
      [req.params.companyId]
    ),
  ]);

  res.json({
    totalCost: parseFloat(total.rows[0].total),
    byAgent: byAgent.rows,
    byDay: byDay.rows,
    period: interval,
  });
});

export default router;

import { describe, it, expect, beforeAll, afterAll, beforeEach } from 'vitest';
import { setupTestDb, teardownTestDb, cleanTables, db } from './setup';

let companyId: string;
let agentId: string;

beforeAll(setupTestDb);
afterAll(teardownTestDb);

beforeEach(async () => {
  await cleanTables();

  const c = await db.query(
    "INSERT INTO companies (name) VALUES ('Test') RETURNING id"
  );
  companyId = c.rows[0].id;

  const a = await db.query(
    `INSERT INTO agents (company_id, name, role, runtime)
     VALUES ($1, 'Alice', 'dev', 'claude') RETURNING id`,
    [companyId]
  );
  agentId = a.rows[0].id;
});

describe('Task Checkout', () => {
  it('should atomically checkout a task', async () => {
    // Create assigned task
    await db.query(
      `INSERT INTO tasks (company_id, title, assigned_to, status)
       VALUES ($1, 'Task 1', $2, 'assigned')`,
      [companyId, agentId]
    );

    // Checkout
    const result = await db.query(
      `UPDATE tasks
       SET locked_by = $1, locked_at = now(), status = 'in_progress'
       WHERE id = (
         SELECT id FROM tasks
         WHERE company_id = $2 AND assigned_to = $1
           AND status = 'assigned' AND locked_by IS NULL
         ORDER BY priority DESC, created_at ASC
         LIMIT 1
         FOR UPDATE SKIP LOCKED
       )
       RETURNING *`,
      [agentId, companyId]
    );

    expect(result.rows).toHaveLength(1);
    expect(result.rows[0].locked_by).toBe(agentId);
    expect(result.rows[0].status).toBe('in_progress');
  });

  it('should not double-checkout the same task', async () => {
    await db.query(
      `INSERT INTO tasks (company_id, title, assigned_to, status, locked_by, locked_at)
       VALUES ($1, 'Task 1', $2, 'in_progress', $2, now())`,
      [companyId, agentId]
    );

    // Try to checkout again
    const result = await db.query(
      `UPDATE tasks
       SET locked_by = $1, locked_at = now(), status = 'in_progress'
       WHERE id = (
         SELECT id FROM tasks
         WHERE company_id = $2 AND assigned_to = $1
           AND status = 'assigned' AND locked_by IS NULL
         LIMIT 1
         FOR UPDATE SKIP LOCKED
       )
       RETURNING *`,
      [agentId, companyId]
    );

    expect(result.rows).toHaveLength(0);
  });
});
packages/server/tests/safety.test.ts
TypeScript
import { describe, it, expect, beforeAll, afterAll, beforeEach } from 'vitest';
import { setupTestDb, teardownTestDb, cleanTables, db } from './setup';

let companyId: string;
let agentId: string;

beforeAll(setupTestDb);
afterAll(teardownTestDb);

beforeEach(async () => {
  await cleanTables();

  const c = await db.query(
    "INSERT INTO companies (name) VALUES ('Test') RETURNING id"
  );
  companyId = c.rows[0].id;

  const a = await db.query(
    `INSERT INTO agents (company_id, name, role, runtime)
     VALUES ($1, 'Alice', 'dev', 'claude') RETURNING id`,
    [companyId]
  );
  agentId = a.rows[0].id;
});

describe('Safety', () => {
  it('should detect delegation depth', async () => {
    // Create chain: task1 → task2 → task3 → task4
    const t1 = await db.query(
      `INSERT INTO tasks (company_id, title, assigned_to, status)
       VALUES ($1, 'T1', $2, 'done') RETURNING id`,
      [companyId, agentId]
    );
    const t2 = await db.query(
      `INSERT INTO tasks (company_id, title, assigned_to, status, parent_id)
       VALUES ($1, 'T2', $2, 'done', $3) RETURNING id`,
      [companyId, agentId, t1.rows[0].id]
    );
    const t3 = await db.query(
      `INSERT INTO tasks (company_id, title, assigned_to, status, parent_id)
       VALUES ($1, 'T3', $2, 'done', $3) RETURNING id`,
      [companyId, agentId, t2.rows[0].id]
    );

    // Walk up the chain
    let depth = 0;
    let currentId = t3.rows[0].id;
    while (currentId) {
      const r = await db.query(
        'SELECT parent_id FROM tasks WHERE id = $1',
        [currentId]
      );
      if (!r.rows[0]?.parent_id) break;
      currentId = r.rows[0].parent_id;
      depth++;
    }

    expect(depth).toBe(2);
  });

  it('should enforce budget limits', async () => {
    const month = new Date().toISOString().slice(0, 7) + '-01';

    await db.query(
      `INSERT INTO budgets (agent_id, month, limit_usd, spent_usd)
       VALUES ($1, $2, 10.00, 9.50)`,
      [agentId, month]
    );

    const budget = await db.query(
      'SELECT * FROM budgets WHERE agent_id = $1 AND month = $2',
      [agentId, month]
    );

    const remaining =
      parseFloat(budget.rows[0].limit_usd) -
      parseFloat(budget.rows[0].spent_usd);

    expect(remaining).toBe(0.50);
    expect(remaining < 1).toBe(true);
  });
});
