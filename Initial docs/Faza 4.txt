CREATE TABLE users (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email       TEXT NOT NULL UNIQUE,
  name        TEXT NOT NULL,
  password_hash TEXT NOT NULL,
  role        TEXT NOT NULL DEFAULT 'member'
    CHECK (role IN ('owner', 'admin', 'member', 'viewer')),
  avatar_url  TEXT,
  last_login  TIMESTAMPTZ,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE user_companies (
  user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  company_id  UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
  role        TEXT NOT NULL DEFAULT 'member'
    CHECK (role IN ('owner', 'admin', 'member', 'viewer')),
  joined_at   TIMESTAMPTZ DEFAULT now(),
  PRIMARY KEY (user_id, company_id)
);

CREATE TABLE api_keys (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  name        TEXT NOT NULL,
  key_hash    TEXT NOT NULL,
  key_prefix  TEXT NOT NULL,           -- first 8 chars for display
  permissions JSONB DEFAULT '["*"]',
  last_used   TIMESTAMPTZ,
  expires_at  TIMESTAMPTZ,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE sessions (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  token_hash  TEXT NOT NULL,
  expires_at  TIMESTAMPTZ NOT NULL,
  ip_address  TEXT,
  user_agent  TEXT,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_sessions_token ON sessions(token_hash);
CREATE INDEX idx_sessions_user ON sessions(user_id);
CREATE INDEX idx_api_keys_hash ON api_keys(key_hash);

import crypto from 'crypto';
import { db } from '../db/client.js';

const TOKEN_EXPIRY_HOURS = 72;
const API_KEY_PREFIX_LENGTH = 8;

export function hashToken(token: string): string {
  return crypto.createHash('sha256').update(token).digest('hex');
}

export function generateToken(): string {
  return crypto.randomBytes(32).toString('hex');
}

export function generateApiKey(): { key: string; prefix: string } {
  const key = `biuro_${crypto.randomBytes(24).toString('hex')}`;
  const prefix = key.slice(0, API_KEY_PREFIX_LENGTH);
  return { key, prefix };
}

export async function hashPassword(password: string): Promise<string> {
  const salt = crypto.randomBytes(16).toString('hex');
  return new Promise((resolve, reject) => {
    crypto.scrypt(password, salt, 64, (err, derivedKey) => {
      if (err) reject(err);
      resolve(`${salt}:${derivedKey.toString('hex')}`);
    });
  });
}

export async function verifyPassword(
  password: string,
  hash: string
): Promise<boolean> {
  const [salt, key] = hash.split(':');
  return new Promise((resolve, reject) => {
    crypto.scrypt(password, salt, 64, (err, derivedKey) => {
      if (err) reject(err);
      resolve(
        crypto.timingSafeEqual(Buffer.from(key, 'hex'), derivedKey)
      );
    });
  });
}

export async function createSession(
  userId: string,
  ip?: string,
  userAgent?: string
): Promise<string> {
  const token = generateToken();
  const tokenHash = hashToken(token);
  const expiresAt = new Date(
    Date.now() + TOKEN_EXPIRY_HOURS * 60 * 60 * 1000
  );

  await db.query(
    `INSERT INTO sessions (user_id, token_hash, expires_at, ip_address, user_agent)
     VALUES ($1, $2, $3, $4, $5)`,
    [userId, tokenHash, expiresAt, ip, userAgent]
  );

  await db.query(
    'UPDATE users SET last_login = now() WHERE id = $1',
    [userId]
  );

  return token;
}

export async function validateSession(
  token: string
): Promise<{ userId: string; email: string; role: string } | null> {
  const tokenHash = hashToken(token);
  const result = await db.query(
    `SELECT s.user_id, u.email, u.role, u.name
     FROM sessions s
     JOIN users u ON u.id = s.user_id
     WHERE s.token_hash = $1 AND s.expires_at > now()`,
    [tokenHash]
  );

  if (result.rows.length === 0) return null;

  return {
    userId: result.rows[0].user_id,
    email: result.rows[0].email,
    role: result.rows[0].role,
  };
}

export async function validateApiKey(
  key: string
): Promise<{ userId: string; permissions: string[] } | null> {
  const keyHash = hashToken(key);
  const result = await db.query(
    `SELECT ak.user_id, ak.permissions
     FROM api_keys ak
     WHERE ak.key_hash = $1
       AND (ak.expires_at IS NULL OR ak.expires_at > now())`,
    [keyHash]
  );

  if (result.rows.length === 0) return null;

  await db.query(
    'UPDATE api_keys SET last_used = now() WHERE key_hash = $1',
    [keyHash]
  );

  return {
    userId: result.rows[0].user_id,
    permissions: result.rows[0].permissions,
  };
}

export async function invalidateSession(token: string): Promise<void> {
  await db.query(
    'DELETE FROM sessions WHERE token_hash = $1',
    [hashToken(token)]
  );
}

export type Role = 'owner' | 'admin' | 'member' | 'viewer';

export interface Permission {
  resource: string;
  actions: string[];
}

const ROLE_PERMISSIONS: Record<Role, Permission[]> = {
  owner: [
    { resource: '*', actions: ['*'] },
  ],
  admin: [
    { resource: 'company', actions: ['read', 'update'] },
    { resource: 'agent', actions: ['*'] },
    { resource: 'task', actions: ['*'] },
    { resource: 'goal', actions: ['*'] },
    { resource: 'tool', actions: ['*'] },
    { resource: 'policy', actions: ['*'] },
    { resource: 'approval', actions: ['*'] },
    { resource: 'budget', actions: ['*'] },
    { resource: 'integration', actions: ['*'] },
    { resource: 'report', actions: ['*'] },
    { resource: 'template', actions: ['read', 'create', 'import'] },
    { resource: 'audit', actions: ['read'] },
    { resource: 'user', actions: ['read'] },
  ],
  member: [
    { resource: 'company', actions: ['read'] },
    { resource: 'agent', actions: ['read', 'create', 'update'] },
    { resource: 'task', actions: ['read', 'create', 'update'] },
    { resource: 'goal', actions: ['read', 'create'] },
    { resource: 'tool', actions: ['read'] },
    { resource: 'policy', actions: ['read'] },
    { resource: 'approval', actions: ['read', 'create'] },
    { resource: 'budget', actions: ['read'] },
    { resource: 'audit', actions: ['read'] },
    { resource: 'report', actions: ['read'] },
  ],
  viewer: [
    { resource: 'company', actions: ['read'] },
    { resource: 'agent', actions: ['read'] },
    { resource: 'task', actions: ['read'] },
    { resource: 'goal', actions: ['read'] },
    { resource: 'tool', actions: ['read'] },
    { resource: 'audit', actions: ['read'] },
    { resource: 'report', actions: ['read'] },
  ],
};

export function hasPermission(
  role: Role,
  resource: string,
  action: string
): boolean {
  const perms = ROLE_PERMISSIONS[role];
  if (!perms) return false;

  return perms.some(
    (p) =>
      (p.resource === '*' || p.resource === resource) &&
      (p.actions.includes('*') || p.actions.includes(action))
  );
}

import { db } from '../db/client.js';
import { hasPermission, type Role } from './roles.js';

/**
 * Check if a user has permission for a resource in a company.
 */
export async function checkAccess(
  userId: string,
  companyId: string,
  resource: string,
  action: string
): Promise<boolean> {
  // Get user's role in this company
  const result = await db.query(
    `SELECT uc.role
     FROM user_companies uc
     WHERE uc.user_id = $1 AND uc.company_id = $2`,
    [userId, companyId]
  );

  if (result.rows.length === 0) {
    // Check global role (owner sees everything they own)
    const user = await db.query(
      'SELECT role FROM users WHERE id = $1',
      [userId]
    );
    if (user.rows[0]?.role === 'owner') return true;
    return false;
  }

  const role = result.rows[0].role as Role;
  return hasPermission(role, resource, action);
}

/**
 * Get all company IDs a user has access to.
 */
export async function getUserCompanies(userId: string): Promise<string[]> {
  const result = await db.query(
    'SELECT company_id FROM user_companies WHERE user_id = $1',
    [userId]
  );
  return result.rows.map((r) => r.company_id);
}
src/auth/middleware.ts
TypeScript
import type { Request, Response, NextFunction } from 'express';
import { validateSession, validateApiKey } from './tokens.js';
import { checkAccess } from './rbac.js';
import { logger } from '../utils/logger.js';

export interface AuthenticatedRequest extends Request {
  user?: {
    userId: string;
    email: string;
    role: string;
  };
}

/**
 * Authentication middleware.
 * Supports Bearer tokens and API keys.
 * If AUTH_ENABLED=false, skips auth (development mode).
 */
export function authenticate(
  req: AuthenticatedRequest,
  res: Response,
  next: NextFunction
) {
  // Dev mode — skip auth
  if (process.env.AUTH_ENABLED !== 'true') {
    req.user = {
      userId: 'dev-user',
      email: 'dev@localhost',
      role: 'owner',
    };
    return next();
  }

  const authHeader = req.headers.authorization;

  if (!authHeader) {
    return res.status(401).json({ error: 'Authentication required' });
  }

  if (authHeader.startsWith('Bearer ')) {
    const token = authHeader.slice(7);

    // Check if it's an API key
    if (token.startsWith('biuro_')) {
      validateApiKey(token).then((result) => {
        if (!result) {
          return res.status(401).json({ error: 'Invalid API key' });
        }
        req.user = {
          userId: result.userId,
          email: '',
          role: 'member',
        };
        next();
      });
    } else {
      // Session token
      validateSession(token).then((result) => {
        if (!result) {
          return res.status(401).json({ error: 'Invalid or expired token' });
        }
        req.user = result;
        next();
      });
    }
  } else {
    return res.status(401).json({ error: 'Invalid auth format' });
  }
}

/**
 * Authorization middleware factory.
 * Usage: authorize('agent', 'create')
 */
export function authorize(resource: string, action: string) {
  return async (
    req: AuthenticatedRequest,
    res: Response,
    next: NextFunction
  ) => {
    if (process.env.AUTH_ENABLED !== 'true') return next();

    const userId = req.user?.userId;
    if (!userId) return res.status(401).json({ error: 'Not authenticated' });

    // Get company ID from route params or body
    const companyId =
      req.params.companyId ||
      req.params.id ||
      req.body?.company_id;

    if (!companyId) {
      // Some routes don't need company-level auth
      return next();
    }

    const allowed = await checkAccess(userId, companyId, resource, action);
    if (!allowed) {
      logger.warn(
        { userId, companyId, resource, action },
        'Access denied'
      );
      return res.status(403).json({
        error: `Insufficient permissions: ${resource}.${action}`,
      });
    }

    next();
  };
}

import { Router } from 'express';
import { db } from '../db/client.js';
import {
  hashPassword,
  verifyPassword,
  createSession,
  invalidateSession,
  generateApiKey,
  hashToken,
} from '../auth/tokens.js';
import { authenticate, type AuthenticatedRequest } from '../auth/middleware.js';
import { z } from 'zod';

const router = Router();

const registerSchema = z.object({
  email: z.string().email(),
  name: z.string().min(1),
  password: z.string().min(8),
});

const loginSchema = z.object({
  email: z.string().email(),
  password: z.string(),
});

// Register
router.post('/register', async (req, res) => {
  const parsed = registerSchema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error });

  const { email, name, password } = parsed.data;

  // Check existing
  const existing = await db.query(
    'SELECT id FROM users WHERE email = $1',
    [email]
  );
  if (existing.rows.length > 0) {
    return res.status(409).json({ error: 'Email already registered' });
  }

  const passwordHash = await hashPassword(password);

  // First user is owner
  const userCount = await db.query('SELECT COUNT(*) FROM users');
  const role = parseInt(userCount.rows[0].count) === 0 ? 'owner' : 'member';

  const result = await db.query(
    `INSERT INTO users (email, name, password_hash, role)
     VALUES ($1, $2, $3, $4)
     RETURNING id, email, name, role`,
    [email, name, passwordHash, role]
  );

  const user = result.rows[0];
  const token = await createSession(
    user.id,
    req.ip,
    req.headers['user-agent']
  );

  res.status(201).json({
    user: { id: user.id, email: user.email, name: user.name, role: user.role },
    token,
  });
});

// Login
router.post('/login', async (req, res) => {
  const parsed = loginSchema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error });

  const { email, password } = parsed.data;

  const result = await db.query(
    'SELECT * FROM users WHERE email = $1',
    [email]
  );
  if (result.rows.length === 0) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const user = result.rows[0];
  const valid = await verifyPassword(password, user.password_hash);
  if (!valid) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const token = await createSession(
    user.id,
    req.ip,
    req.headers['user-agent']
  );

  res.json({
    user: { id: user.id, email: user.email, name: user.name, role: user.role },
    token,
  });
});

// Logout
router.post('/logout', authenticate, async (req: AuthenticatedRequest, res) => {
  const token = req.headers.authorization?.slice(7);
  if (token) await invalidateSession(token);
  res.json({ ok: true });
});

// Me
router.get('/me', authenticate, async (req: AuthenticatedRequest, res) => {
  const result = await db.query(
    'SELECT id, email, name, role, last_login, created_at FROM users WHERE id = $1',
    [req.user!.userId]
  );
  if (result.rows.length === 0) return res.status(404).json({ error: 'User not found' });

  const companies = await db.query(
    `SELECT c.*, uc.role as user_role
     FROM user_companies uc
     JOIN companies c ON c.id = uc.company_id
     WHERE uc.user_id = $1`,
    [req.user!.userId]
  );

  res.json({
    ...result.rows[0],
    companies: companies.rows,
  });
});

// Create API key
router.post('/api-keys', authenticate, async (req: AuthenticatedRequest, res) => {
  const { name, permissions } = req.body;
  const { key, prefix } = generateApiKey();
  const keyHash = hashToken(key);

  await db.query(
    `INSERT INTO api_keys (user_id, name, key_hash, key_prefix, permissions)
     VALUES ($1, $2, $3, $4, $5)`,
    [
      req.user!.userId,
      name || 'API Key',
      keyHash,
      prefix,
      JSON.stringify(permissions || ['*']),
    ]
  );

  // Return key ONCE — never stored in plaintext
  res.status(201).json({
    key,
    prefix,
    note: 'Save this key — it will not be shown again.',
  });
});

// List API keys
router.get('/api-keys', authenticate, async (req: AuthenticatedRequest, res) => {
  const result = await db.query(
    `SELECT id, name, key_prefix, permissions, last_used, created_at
     FROM api_keys WHERE user_id = $1
     ORDER BY created_at DESC`,
    [req.user!.userId]
  );
  res.json(result.rows);
});

// Revoke API key
router.delete('/api-keys/:id', authenticate, async (req: AuthenticatedRequest, res) => {
  await db.query(
    'DELETE FROM api_keys WHERE id = $1 AND user_id = $2',
    [req.params.id, req.user!.userId]
  );
  res.json({ deleted: true });
});

// Notifications
router.get('/notifications', authenticate, async (req: AuthenticatedRequest, res) => {
  const result = await db.query(
    `SELECT * FROM notifications
     WHERE user_id = $1
     ORDER BY created_at DESC
     LIMIT 50`,
    [req.user!.userId]
  );
  res.json(result.rows);
});

router.post('/notifications/mark-read', authenticate, async (req: AuthenticatedRequest, res) => {
  const { ids } = req.body;
  if (ids && ids.length > 0) {
    await db.query(
      'UPDATE notifications SET read = true WHERE id = ANY($1) AND user_id = $2',
      [ids, req.user!.userId]
    );
  } else {
    await db.query(
      'UPDATE notifications SET read = true WHERE user_id = $1',
      [req.user!.userId]
    );
  }
  res.json({ ok: true });
});

export default router;

/**
 * Export a company as a reusable template.
 */
export async function exportCompany(
  companyId: string
): Promise<CompanyTemplate> {
  const company = await db.query(
    'SELECT * FROM companies WHERE id = $1',
    [companyId]
  );
  if (company.rows.length === 0) throw new Error('Company not found');

  const c = company.rows[0];

  // Goals
  const goals = await db.query(
    'SELECT * FROM goals WHERE company_id = $1 ORDER BY created_at ASC',
    [companyId]
  );

  // Agents
  const agents = await db.query(
    'SELECT * FROM agents WHERE company_id = $1 AND status != $2 ORDER BY created_at ASC',
    [companyId, 'terminated']
  );

  // Budgets
  const budgets = await db.query(
    `SELECT b.agent_id, b.limit_usd
     FROM budgets b
     JOIN agents a ON a.id = b.agent_id
     WHERE a.company_id = $1
       AND b.month = date_trunc('month', now())::date`,
    [companyId]
  );
  const budgetMap = new Map(
    budgets.rows.map((b) => [b.agent_id, parseFloat(b.limit_usd)])
  );

  // Tools
  const tools = await db.query(
    'SELECT * FROM tools WHERE company_id = $1 ORDER BY name ASC',
    [companyId]
  );

  // Agent-tool assignments
  const agentTools = await db.query(
    `SELECT at.*, t.name as tool_name
     FROM agent_tools at
     JOIN tools t ON t.id = at.tool_id
     JOIN agents a ON a.id = at.agent_id
     WHERE a.company_id = $1`,
    [companyId]
  );

  // Policies
  const policies = await db.query(
    'SELECT * FROM policies WHERE company_id = $1',
    [companyId]
  );

  // Integrations (schema only)
  const integrations = await db.query(
    'SELECT type, name, events FROM integrations WHERE company_id = $1',
    [companyId]
  );

  // Build ref maps (UUID → ref string)
  const goalRefMap = new Map<string, string>();
  goals.rows.forEach((g, i) => goalRefMap.set(g.id, `goal_${i}`));

  const agentRefMap = new Map<string, string>();
  agents.rows.forEach((a, i) => agentRefMap.set(a.id, `agent_${i}`));

  const toolRefMap = new Map<string, string>();
  tools.rows.forEach((t, i) => toolRefMap.set(t.id, `tool_${i}`));

  const template: CompanyTemplate = {
    version: '1.0.0',
    name: c.name,
    description: c.mission || '',
    exported_at: new Date().toISOString(),
    company: {
      name: c.name,
      mission: c.mission || '',
    },
    goals: goals.rows.map((g) => ({
      ref: goalRefMap.get(g.id)!,
      parent_ref: g.parent_id ? goalRefMap.get(g.parent_id) || null : null,
      title: g.title,
      description: g.description || '',
    })),
    agents: agents.rows.map((a) => ({
      ref: agentRefMap.get(a.id)!,
      name: a.name,
      role: a.role,
      title: a.title || '',
      reports_to_ref: a.reports_to
        ? agentRefMap.get(a.reports_to) || null
        : null,
      runtime: a.runtime,
      model: a.model,
      system_prompt: a.system_prompt || '',
      config: a.config || {},
      monthly_budget_usd: budgetMap.get(a.id) || null,
    })),
    tools: tools.rows.map((t) => ({
      ref: toolRefMap.get(t.id)!,
      name: t.name,
      type: t.type,
      description: t.description || '',
      config: {}, // SCRUBBED — no secrets
      schema: t.schema || {},
    })),
    agent_tools: agentTools.rows.map((at) => ({
      agent_ref: agentRefMap.get(at.agent_id)!,
      tool_ref: toolRefMap.get(at.tool_id)!,
      permissions: at.permissions,
      rate_limit: at.rate_limit,
    })),
    policies: policies.rows.map((p) => ({
      name: p.name,
      description: p.description || '',
      type: p.type,
      rules: p.rules,
    })),
    integrations: integrations.rows.map((i) => ({
      type: i.type,
      name: i.name,
      events: i.events,
      config_schema: {},
    })),
  };

  // Sanitize
  return sanitizeTemplate(template);
}

import type { CompanyTemplate } from './exporter.js';

const SENSITIVE_KEYS = [
  'api_key',
  'apiKey',
  'api_secret',
  'secret',
  'token',
  'password',
  'auth',
  'authorization',
  'bearer',
  'credential',
  'private_key',
  'access_token',
  'refresh_token',
  'webhook_url',
];

/**
 * Deep-scrub sensitive values from template data.
 */
export function sanitizeTemplate(
  template: CompanyTemplate
): CompanyTemplate {
  return JSON.parse(JSON.stringify(template), (key, value) => {
    if (typeof key === 'string' && typeof value === 'string') {
      const lowerKey = key.toLowerCase();
      if (SENSITIVE_KEYS.some((sk) => lowerKey.includes(sk))) {
        return '<<REDACTED>>';
      }
    }
    return value;
  });
}

/**
 * Validate a template before import.
 */
export function validateTemplate(
  data: any
): { valid: boolean; errors: string[] } {
  const errors: string[] = [];

  if (!data.version) errors.push('Missing version');
  if (!data.company?.name) errors.push('Missing company name');
  if (!Array.isArray(data.agents)) errors.push('Missing agents array');
  if (!Array.isArray(data.goals)) errors.push('Missing goals array');

  // Check ref integrity
  const agentRefs = new Set(data.agents?.map((a: any) => a.ref) || []);
  const goalRefs = new Set(data.goals?.map((g: any) => g.ref) || []);
  const toolRefs = new Set(data.tools?.map((t: any) => t.ref) || []);

  for (const agent of data.agents || []) {
    if (agent.reports_to_ref && !agentRefs.has(agent.reports_to_ref)) {
      errors.push(
        `Agent "${agent.name}" references unknown reports_to: ${agent.reports_to_ref}`
      );
    }
  }

  for (const goal of data.goals || []) {
    if (goal.parent_ref && !goalRefs.has(goal.parent_ref)) {
      errors.push(
        `Goal "${goal.title}" references unknown parent: ${goal.parent_ref}`
      );
    }
  }

  for (const at of data.agent_tools || []) {
    if (!agentRefs.has(at.agent_ref))
      errors.push(`agent_tool references unknown agent: ${at.agent_ref}`);
    if (!toolRefs.has(at.tool_ref))
      errors.push(`agent_tool references unknown tool: ${at.tool_ref}`);
  }

  return { valid: errors.length === 0, errors };
}

import { db } from '../db/client.js';
import { validateTemplate } from './sanitizer.js';
import { logger } from '../utils/logger.js';
import type { CompanyTemplate } from './exporter.js';

/**
 * Import a template into a new company.
 * Handles ref→UUID mapping and collision detection.
 */
export async function importTemplate(
  template: CompanyTemplate,
  overrides?: {
    companyName?: string;
    mission?: string;
  }
): Promise<{ companyId: string; summary: any }> {
  // Validate
  const validation = validateTemplate(template);
  if (!validation.valid) {
    throw new Error(`Invalid template: ${validation.errors.join(', ')}`);
  }

  return db.transaction(async (client) => {
    // 1. Create company
    const companyName =
      overrides?.companyName || template.company.name;

    // Handle name collision
    const existing = await client.query(
      'SELECT id FROM companies WHERE name = $1',
      [companyName]
    );
    const finalName =
      existing.rows.length > 0
        ? `${companyName} (${new Date().toISOString().slice(0, 10)})`
        : companyName;

    const company = await client.query(
      `INSERT INTO companies (name, mission)
       VALUES ($1, $2) RETURNING id`,
      [finalName, overrides?.mission || template.company.mission]
    );
    const companyId = company.rows[0].id;

    // 2. Create goals (respecting parent refs)
    const goalIdMap = new Map<string, string>();

    // Sort: parents first
    const sortedGoals = [...template.goals].sort((a, b) => {
      if (!a.parent_ref && b.parent_ref) return -1;
      if (a.parent_ref && !b.parent_ref) return 1;
      return 0;
    });

    for (const goal of sortedGoals) {
      const parentId = goal.parent_ref
        ? goalIdMap.get(goal.parent_ref) || null
        : null;

      const result = await client.query(
        `INSERT INTO goals (company_id, parent_id, title, description)
         VALUES ($1, $2, $3, $4) RETURNING id`,
        [companyId, parentId, goal.title, goal.description]
      );
      goalIdMap.set(goal.ref, result.rows[0].id);
    }

    // 3. Create agents (respecting reports_to refs)
    const agentIdMap = new Map<string, string>();

    // Sort: managers first
    const sortedAgents = [...template.agents].sort((a, b) => {
      if (!a.reports_to_ref && b.reports_to_ref) return -1;
      if (a.reports_to_ref && !b.reports_to_ref) return 1;
      return 0;
    });

    for (const agent of sortedAgents) {
      const reportsTo = agent.reports_to_ref
        ? agentIdMap.get(agent.reports_to_ref) || null
        : null;

      const result = await client.query(
        `INSERT INTO agents
          (company_id, name, role, title, reports_to, runtime, model, system_prompt, config)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
         RETURNING id`,
        [
          companyId,
          agent.name,
          agent.role,
          agent.title,
          reportsTo,
          agent.runtime,
          agent.model,
          agent.system_prompt,
          JSON.stringify(agent.config),
        ]
      );
      const agentId = result.rows[0].id;
      agentIdMap.set(agent.ref, agentId);

      // Budget
      if (agent.monthly_budget_usd) {
        const month = new Date().toISOString().slice(0, 7) + '-01';
        await client.query(
          `INSERT INTO budgets (agent_id, month, limit_usd)
           VALUES ($1, $2, $3)`,
          [agentId, month, agent.monthly_budget_usd]
        );
      }
    }

    // 4. Create tools
    const toolIdMap = new Map<string, string>();

    for (const tool of template.tools) {
      const result = await client.query(
        `INSERT INTO tools (company_id, name, type, description, config, schema)
         VALUES ($1, $2, $3, $4, $5, $6)
         RETURNING id`,
        [
          companyId,
          tool.name,
          tool.type,
          tool.description,
          JSON.stringify(tool.config),
          JSON.stringify(tool.schema),
        ]
      );
      toolIdMap.set(tool.ref, result.rows[0].id);
    }

    // 5. Create agent-tool assignments
    for (const at of template.agent_tools) {
      const agentId = agentIdMap.get(at.agent_ref);
      const toolId = toolIdMap.get(at.tool_ref);
      if (agentId && toolId) {
        await client.query(
          `INSERT INTO agent_tools (agent_id, tool_id, permissions, rate_limit)
           VALUES ($1, $2, $3, $4)`,
          [agentId, toolId, JSON.stringify(at.permissions), at.rate_limit]
        );
      }
    }

    // 6. Create policies
    for (const policy of template.policies) {
      await client.query(
        `INSERT INTO policies (company_id, name, description, type, rules)
         VALUES ($1, $2, $3, $4, $5)`,
        [
          companyId,
          policy.name,
          policy.description,
          policy.type,
          JSON.stringify(policy.rules),
        ]
      );
    }

    // Audit
    await client.query(
      `INSERT INTO audit_log (company_id, action, entity_type, entity_id, details)
       VALUES ($1, 'template.imported', 'company', $1, $2)`,
      [
        companyId,
        JSON.stringify({
          template_name: template.name,
          template_version: template.version,
          agents: template.agents.length,
          goals: template.goals.length,
          tools: template.tools.length,
          policies: template.policies.length,
        }),
      ]
    );

    const summary = {
      companyId,
      companyName: finalName,
      goals: goalIdMap.size,
      agents: agentIdMap.size,
      tools: toolIdMap.size,
      policies: template.policies.length,
      agentTools: template.agent_tools.length,
    };

    logger.info(summary, '📦 Template imported');

    return { companyId, summary };
  });
}

type IntegrationSender = (
  config: any,
  event: string,
  payload: any

const senders: Record<string, IntegrationSender> = {
  slack: sendSlack,
  discord: sendDiscord,
  email: sendEmail,
  webhook: sendWebhook,

/**
 * Dispatch an event to all matching integrations for a company.
 */
export async function dispatchEvent(
  companyId: string,
  event: string,
  payload: any
): Promise<void> {
  const integrations = await db.query(
    `SELECT * FROM integrations
     WHERE company_id = $1
       AND enabled = true
       AND ($2 = ANY(events) OR events = '{}')`,
    [companyId, event]
  );

  for (const integration of integrations.rows) {
    const sender = senders[integration.type];
    if (!sender) {
      logger.warn({ type: integration.type }, 'Unknown integration type');
      continue;
    }

    try {
      await sender(integration.config, event, payload);

      await db.query(
        `UPDATE integrations SET last_triggered = now(), error_count = 0 WHERE id = $1`,
        [integration.id]
      );

      await db.query(
        `INSERT INTO integration_log (integration_id, event_type, payload, status)
         VALUES ($1, $2, $3, 'sent')`,
        [integration.id, event, JSON.stringify(payload)]
      );
    } catch (err: any) {
      logger.error(
        { err, integration: integration.name, event },
        'Integration dispatch failed'
      );

      await db.query(
        `UPDATE integrations SET error_count = error_count + 1 WHERE id = $1`,
        [integration.id]
      );

      await db.query(
        `INSERT INTO integration_log (integration_id, event_type, payload, status, error)
         VALUES ($1, $2, $3, 'failed', $4)`,
        [integration.id, event, JSON.stringify(payload), err.message]
      );

      // Auto-disable after 10 consecutive errors
      if (integration.error_count + 1 >= 10) {
        await db.query(
          'UPDATE integrations SET enabled = false WHERE id = $1',
          [integration.id]
        );
        logger.warn(
          { integration: integration.name },
          'Integration auto-disabled after 10 errors'
        );
      }
    }
  }
}

export async function sendSlack(

        title: `🏢 ${event}`,

      },

  };

    headers: { 'Content-Type': 'application/json' },

function formatPayload(event: string, payload: any): string {
  switch (event) {

    case 'agent.auto_paused':
      return `⚠️ *${payload.agentName}* auto-paused: ${payload.reason}`;
    default:
      return JSON.stringify(payload, null, 2).slice(0, 300);
  }

export async function sendWebhook(
  config: {
    url: string;
    secret?: string;
    headers?: Record<string, string>;
  },
  event: string,
  payload: any
): Promise<void> {
  if (!config.url) throw new Error('Missing webhook URL');

  const body = JSON.stringify({
    event,
    payload,
    timestamp: new Date().toISOString(),
  });

  const headers: Record<string, string> = {
    'Content-Type': 'application/json',
    ...config.headers,
  };

  // HMAC signature for webhook verification
  if (config.secret) {
    const crypto = await import('crypto');
    const signature = crypto
      .createHmac('sha256', config.secret)
      .update(body)
      .digest('hex');
    headers['X-Biuro-Signature'] = `sha256=${signature}`;
  }

  const res = await fetch(config.url, {
    method: 'POST',
    headers,
    body,
  });

  if (!res.ok) {
    throw new Error(`Webhook error: ${res.status} ${await res.text()}`);
  }
}

  if (!config || typeof config !== 'object') return config;

        key.toLowerCase().includes(s)
      )

export interface ReportData {
  period: { start: string; end: string };
  company: { name: string; mission: string };
  summary: {
    totalCost: number;
    tasksCreated: number;
    tasksCompleted: number;
    heartbeats: number;
    toolCalls: number;
    approvals: { pending: number; approved: number; rejected: number };
  };
  agents: Array<{
    name: string;
    role: string;
    status: string;
    heartbeats: number;
    tasksCompleted: number;
    cost: number;
    budgetUsedPct: number;
  }>;
  topTasks: Array<{
    title: string;
    status: string;
    assignedTo: string;
    result: string | null;
  }>;
  issues: string[];
}

export async function generateReportData(
  companyId: string,
  periodDays: number = 1
): Promise<ReportData> {
  const interval = `${periodDays} days`;

  const company = await db.query(
    'SELECT * FROM companies WHERE id = $1',
    [companyId]
  );
  const c = company.rows[0];

  // Summary metrics
  const [costs, tasksCreated, tasksCompleted, heartbeats, toolCalls, approvals] =
    await Promise.all([
      db.query(
        `SELECT COALESCE(SUM(cost_usd), 0) as total
         FROM audit_log WHERE company_id = $1 AND created_at > now() - interval '${interval}'`,
        [companyId]
      ),
      db.query(
        `SELECT COUNT(*) as count FROM tasks
         WHERE company_id = $1 AND created_at > now() - interval '${interval}'`,
        [companyId]
      ),
      db.query(
        `SELECT COUNT(*) as count FROM tasks
         WHERE company_id = $1 AND status = 'done'
           AND updated_at > now() - interval '${interval}'`,
        [companyId]
      ),
      db.query(
        `SELECT COUNT(*) as count FROM heartbeats h
         JOIN agents a ON a.id = h.agent_id
         WHERE a.company_id = $1 AND h.created_at > now() - interval '${interval}'`,
        [companyId]
      ),
      db.query(
        `SELECT COUNT(*) as count FROM tool_calls
         WHERE company_id = $1 AND created_at > now() - interval '${interval}'`,
        [companyId]
      ),
      db.query(
        `SELECT status, COUNT(*) as count FROM approvals
         WHERE company_id = $1 AND created_at > now() - interval '${interval}'
         GROUP BY status`,
        [companyId]
      ),
    ]);

  const approvalCounts = {
    pending: 0,
    approved: 0,
    rejected: 0,
  };
  for (const row of approvals.rows) {
    approvalCounts[row.status as keyof typeof approvalCounts] =
      parseInt(row.count);
  }

  // Per-agent metrics
  const agents = await db.query(
    `SELECT
       a.name, a.role, a.status,
       COALESCE(h.heartbeats, 0) as heartbeats,
       COALESCE(tc.tasks_completed, 0) as tasks_completed,
       COALESCE(al.cost, 0) as cost,
       b.limit_usd as budget_limit,
       b.spent_usd as budget_spent
     FROM agents a
     LEFT JOIN (
       SELECT agent_id, COUNT(*) as heartbeats FROM heartbeats
       WHERE created_at > now() - interval '${interval}'
       GROUP BY agent_id
     ) h ON h.agent_id = a.id
     LEFT JOIN (
       SELECT assigned_to, COUNT(*) as tasks_completed FROM tasks
       WHERE status = 'done' AND updated_at > now() - interval '${interval}'
       GROUP BY assigned_to
     ) tc ON tc.assigned_to = a.id
     LEFT JOIN (
       SELECT agent_id, SUM(cost_usd) as cost FROM audit_log
       WHERE created_at > now() - interval '${interval}' AND cost_usd > 0
       GROUP BY agent_id
     ) al ON al.agent_id = a.id
     LEFT JOIN budgets b ON b.agent_id = a.id
       AND b.month = date_trunc('month', now())::date
     WHERE a.company_id = $1 AND a.status != 'terminated'
     ORDER BY cost DESC`,
    [companyId]
  );

  // Recent completed tasks
  const topTasks = await db.query(
    `SELECT t.title, t.status, a.name as assigned_to, t.result
     FROM tasks t
     LEFT JOIN agents a ON a.id = t.assigned_to
     WHERE t.company_id = $1 AND t.status = 'done'
       AND t.updated_at > now() - interval '${interval}'
     ORDER BY t.updated_at DESC
     LIMIT 10`,
    [companyId]
  );

  // Detect issues
  const issues: string[] = [];

  // Paused agents
  const paused = agents.rows.filter((a) => a.status === 'paused');
  if (paused.length > 0) {
    issues.push(
      `⚠️ ${paused.length} agent(s) paused: ${paused.map((a) => a.name).join(', ')}`
    );
  }

  // Budget warnings
  for (const a of agents.rows) {
    if (a.budget_limit && a.budget_spent) {
      const pct =
        (parseFloat(a.budget_spent) / parseFloat(a.budget_limit)) * 100;
      if (pct > 90) {
        issues.push(
          `🔴 ${a.name} budget ${pct.toFixed(0)}% used ($${parseFloat(a.budget_spent).toFixed(2)} / $${parseFloat(a.budget_limit).toFixed(2)})`
        );
      }
    }
  }

  // Blocked tasks
  const blocked = await db.query(
    `SELECT COUNT(*) as count FROM tasks
     WHERE company_id = $1 AND status = 'blocked'`,
    [companyId]
  );
  if (parseInt(blocked.rows[0].count) > 0) {
    issues.push(
      `🚫 ${blocked.rows[0].count} task(s) currently blocked`
    );
  }

  // Pending approvals
  if (approvalCounts.pending > 0) {
    issues.push(
      `🎫 ${approvalCounts.pending} approval(s) pending your decision`
    );
  }

  return {
    period: {
      start: new Date(
        Date.now() - periodDays * 24 * 60 * 60 * 1000
      ).toISOString(),
      end: new Date().toISOString(),
    },
    company: { name: c.name, mission: c.mission || '' },
    summary: {
      totalCost: parseFloat(costs.rows[0].total),
      tasksCreated: parseInt(tasksCreated.rows[0].count),
      tasksCompleted: parseInt(tasksCompleted.rows[0].count),
      heartbeats: parseInt(heartbeats.rows[0].count),
      toolCalls: parseInt(toolCalls.rows[0].count),
      approvals: approvalCounts,
    },
    agents: agents.rows.map((a) => ({
      name: a.name,
      role: a.role,
      status: a.status,
      heartbeats: parseInt(a.heartbeats),
      tasksCompleted: parseInt(a.tasks_completed),
      cost: parseFloat(a.cost),
      budgetUsedPct:
        a.budget_limit && parseFloat(a.budget_limit) > 0
          ? (parseFloat(a.budget_spent || 0) /
              parseFloat(a.budget_limit)) *
            100
          : 0,
    })),
    topTasks: topTasks.rows.map((t) => ({
      title: t.title,
      status: t.status,
      assignedTo: t.assigned_to || 'Unassigned',
      result: t.result,
    })),
    issues,
  };
}

  if (topTasks.length > 0) {

    md += `\n`;
  }

  md += `---\n*Generated by Autonomiczne Biuro at ${new Date().toISOString()}*\n`;

async function runReport(report: any): Promise<void> {
  logger.info(
    { reportId: report.id, type: report.type },

  );

  // Determine period

  if (report.type === 'cost_report') periodDays = 30;

  // Generate data
  const data = await generateReportData(report.company_id, periodDays);

  // Format
  let content: string;
  switch (report.type) {
    case 'daily_summary':
      content = formatDailySummary(data);
      break;
    case 'weekly_report':
      content = formatDailySummary(data); // reuse for now
      break;
    case 'cost_report':
      content = formatCostReport(data);
      break;
    default:
      content = JSON.stringify(data, null, 2);
  }

  // Save to history
  await db.query(
    `INSERT INTO report_history (report_id, content, format, metadata)
     VALUES ($1, $2, $3, $4)`,
    [
      report.id,
      content,
      report.format,
      JSON.stringify({ period: data.period, summary: data.summary }),
    ]
  );

  // Dispatch to integrations
  await dispatchEvent(report.company_id, 'report.generated', {
    reportName: report.name,
    type: report.type,
    content,
    summary: data.summary,
  });

  // Update next_run based on schedule (simplified cron)
  const nextRun = calculateNextRun(report.schedule);

  await db.query(
    `UPDATE scheduled_reports SET last_run = now(), next_run = $2 WHERE id = $1`,
    [report.id, nextRun]
  );

  logger.info(
    { reportId: report.id, nextRun },
    '📊 Report completed'
  );
}

function formatCostReport(data: any): string {
  let md = `# 💰 Cost Report — ${data.company.name}\n\n`;
  md += `Total spend: **$${data.summary.totalCost.toFixed(2)}**\n\n`;
  md += `| Agent | Cost | Budget Used |\n|---|---|---|\n`;
  for (const a of data.agents) {
    md += `| ${a.name} | $${a.cost.toFixed(2)} | ${a.budgetUsedPct.toFixed(0)}% |\n`;
  }
  return md;
}

function calculateNextRun(schedule: string): Date {
  // Simplified: supports 'daily', 'weekly', 'hourly', or hour offset
  const now = new Date();

  if (schedule === 'hourly' || schedule === '0 * * * *') {
    return new Date(now.getTime() + 60 * 60 * 1000);
  }
  if (schedule === 'daily' || schedule.startsWith('0 9')) {
    const next = new Date(now);
    next.setDate(next.getDate() + 1);
    next.setHours(9, 0, 0, 0);
    return next;
  }
  if (schedule === 'weekly' || schedule.startsWith('0 9 * * 1')) {
    const next = new Date(now);
    next.setDate(next.getDate() + (((1 + 7 - next.getDay()) % 7) || 7));
    next.setHours(9, 0, 0, 0);
    return next;
  }

  // Default: next day 9 AM
  const next = new Date(now);
  next.setDate(next.getDate() + 1);
  next.setHours(9, 0, 0, 0);
  return next;
}

const DEFAULT_HOST = 'http://localhost:3100';

export function getApiUrl(): string {
  return process.env.BIURO_API_URL || DEFAULT_HOST;
}

export async function api<T = any>(
  path: string,
  options?: RequestInit
): Promise<T> {
  const url = `${getApiUrl()}/api${path}`;
  const token = process.env.BIURO_TOKEN || '';

  const res = await fetch(url, {
    headers: {
      'Content-Type': 'application/json',
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
    },
    ...options,
  });

  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: res.statusText }));
    throw new Error(err.error || `HTTP ${res.status}`);
  }

  return res.json();
}

export function status(s: string): string {
  const colors: Record<string, (s: string) => string> = {
    idle: chalk.gray,
    working: chalk.green,
    paused: chalk.yellow,
    terminated: chalk.red,
    backlog: chalk.gray,
    assigned: chalk.blue,
    in_progress: chalk.green,
    done: chalk.greenBright,
    blocked: chalk.red,
    pending: chalk.yellow,
    approved: chalk.green,
    rejected: chalk.red,
  };
  const colorFn = colors[s] || chalk.white;
  return colorFn(s);
}

export function cost(amount: number): string {
  if (amount > 10) return chalk.red(`$${amount.toFixed(4)}`);
  if (amount > 1) return chalk.yellow(`$${amount.toFixed(4)}`);
  return chalk.green(`$${amount.toFixed(4)}`);
}

        );
        for (const log of logs.reverse()) {
          const icon = ACTION_ICONS[log.action] || '📝';
          const time = new Date(log.created_at).toLocaleTimeString();
          const agent = log.agent_name
            ? chalk.blue(log.agent_name)
            : chalk.gray('system');
          const costStr =
            log.cost_usd && parseFloat(log.cost_usd) > 0
              ? chalk.yellow(` $${parseFloat(log.cost_usd).toFixed(4)}`)
              : '';

          console.log(
            `${chalk.gray(time)} ${icon} ${chalk.white(log.action)} ${agent}${costStr}`
          );
        }
      };

      await printLogs();

      if (opts.follow) {
        console.log(chalk.gray('\n--- Following (Ctrl+C to stop) ---\n'));
        setInterval(printLogs, 5000);
      }
    });
}

function CreateIntegrationModal({
  open,
  onClose,
  onCreated,
  companyId,
}: {
  open: boolean;
  onClose: () => void;
  onCreated: () => void;
  companyId: string;
}) {
  const [type, setType] = useState('slack');
  const [name, setName] = useState('');
  const [configJson, setConfigJson] = useState('{"webhook_url": ""}');
  const [events, setEvents] = useState('');
  const [loading, setLoading] = useState(false);

  const configTemplates: Record<string, string> = {
    slack: '{"webhook_url": "https://hooks.slack.com/services/..."}',
    discord: '{"webhook_url": "https://discord.com/api/webhooks/..."}',
    email: '{"service": "resend", "api_key": "re_...", "to": "you@example.com"}',
    webhook:
      '{"url": "https://your-endpoint.com/webhook", "secret": "your-hmac-secret"}',
  };

  const handleTypeChange = (t: string) => {
    setType(t);
    setConfigJson(configTemplates[t] || '{}');
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);
    try {
      await fetch(`/api/companies/${companyId}/integrations`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          type,
          name,
          config: JSON.parse(configJson),
          events: events
            ? events.split(',').map((e) => e.trim())
            : [],
        }),
      });
      onCreated();
      onClose();
    } finally {
      setLoading(false);
    }
  };

  return (
    <Modal open={open} onClose={onClose} title="Add Integration" wide>
      <form onSubmit={handleSubmit} className="space-y-4">
        <div className="grid grid-cols-2 gap-4">

              { value: 'webhook', label: '🔗 Webhook' },

            label="Name"
            value={name}
            onChange={(e) => setName(e.target.value)}
            placeholder="My Slack notifications"
            required
          />
        </div>
        <Textarea
          label="Config (JSON)"
          value={configJson}
          onChange={(e) => setConfigJson(e.target.value)}
          rows={4}
          className="font-mono text-xs"
        />
        <Input
          label="Events (comma-separated, leave empty for all)"

          onChange={(e) => setEvents(e.target.value)}
          placeholder="heartbeat.completed, task.completed, approval.requested"

            Cancel
          </Button>
          <Button type="submit" loading={loading}>
            🔗 Connect
          </Button>
        </div>
      </form>
    </Modal>
  );
}
