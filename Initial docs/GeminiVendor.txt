4. Nowy plik: packages/server/src/runtimes/gemini.ts
Utwórz plik: packages/server/src/runtimes/gemini.ts

TypeScript

import { GoogleGenAI } from '@google/genai';
import { env } from '../env.js';
import { logger } from '../utils/logger.js';
import type {
  AgentRuntime,
  AgentContext,
  AgentResponse,
  AgentAction,
} from './types.js';

// Gemini 2.5 Flash pricing (per million tokens)
const COST_PER_MTK_INPUT = 0.15;     // $0.15 / 1M input tokens (≤200k)
const COST_PER_MTK_OUTPUT = 0.60;    // $0.60 / 1M output tokens (non-thinking)
// Powyżej 200k kontekstu: input $0.35, output $1.50
// Thinking tokens: output $3.50
// Dostosuj jeśli używasz innego modelu

export class GeminiRuntime implements AgentRuntime {
  name = 'gemini';
  private client: GoogleGenAI;
  private model: string;

  constructor(model?: string) {
    if (!env.GOOGLE_API_KEY) {
      throw new Error('GOOGLE_API_KEY is required for Gemini runtime');
    }
    this.client = new GoogleGenAI({ apiKey: env.GOOGLE_API_KEY });
    this.model = model || 'gemini-2.5-flash';
  }

  async execute(context: AgentContext): Promise<AgentResponse> {
    const actionSchema = `
Respond with a JSON object containing your actions. Format:
{
  "thinking": "your reasoning about what to do",
  "actions": [
    {
      "type": "complete_task",
      "result": "description of what you accomplished"
    }
  ]
}

Available action types:
- complete_task: { "type": "complete_task", "result": "..." }
- delegate: { "type": "delegate", "to": "agent_role", "description": "..." }
- message: { "type": "message", "to": "agent_role", "content": "..." }
- use_tool: { "type": "use_tool", "tool": "tool_name", "params": { ... } }
- blocked: { "type": "blocked", "reason": "..." }
- continue: { "type": "continue", "result": "progress so far..." }

You may include multiple actions. Always include at least one.
Respond ONLY with valid JSON.`;

    // Build contents array for Gemini
    const contents: Array<{ role: string; parts: Array<{ text: string }> }> = [];

    // Add conversation history
    for (const msg of context.conversationHistory) {
      contents.push({
        role: msg.role === 'assistant' ? 'model' : 'user',
        parts: [{ text: msg.content }],
      });
    }

    // Add current task + action schema
    contents.push({
      role: 'user',
      parts: [{ text: `${context.taskDescription}\n\n${actionSchema}` }],
    });

    try {
      const response = await this.client.models.generateContent({
        model: this.model,
        contents,
        config: {
          systemInstruction: context.systemPrompt,
          maxOutputTokens: 4096,
          temperature: 0.7,
          responseMimeType: 'application/json',
        },
      });

      const rawText = response.text || '';

      // Parse usage metadata
      const usageMetadata = response.usageMetadata;
      const inputTokens = usageMetadata?.promptTokenCount || 0;
      const outputTokens = usageMetadata?.candidatesTokenCount || 0;

      // Calculate cost
      const costUsd =
        (inputTokens / 1_000_000) * COST_PER_MTK_INPUT +
        (outputTokens / 1_000_000) * COST_PER_MTK_OUTPUT;

      // Parse actions from JSON response
      let actions: AgentAction[] = [];
      try {
        const jsonMatch = rawText.match(/\{[\s\S]*\}/);
        if (jsonMatch) {
          const parsed = JSON.parse(jsonMatch[0]);
          actions = parsed.actions || [];
        }
      } catch (err) {
        logger.warn(
          { rawText: rawText.slice(0, 200) },
          'Failed to parse Gemini JSON, treating as complete_task'
        );
        actions = [{ type: 'complete_task', result: rawText }];
      }

      // Validate actions
      actions = actions.filter((a) =>
        [
          'complete_task',
          'delegate',
          'message',
          'use_tool',
          'blocked',
          'continue',
          'request_approval',
        ].includes(a.type)
      );

      if (actions.length === 0) {
        actions = [{ type: 'complete_task', result: rawText }];
      }

      logger.debug(
        {
          model: this.model,
          inputTokens,
          outputTokens,
          cost: costUsd.toFixed(6),
          actions: actions.map((a) => a.type),
        },
        'Gemini response received'
      );

      return {
        actions,
        rawResponse: rawText,
        cost: {
          input_tokens: inputTokens,
          output_tokens: outputTokens,
          usd: costUsd,
        },
      };
    } catch (err: any) {
      // Handle Gemini-specific errors
      if (err.status === 429) {
        logger.warn('Gemini rate limit hit, will retry next heartbeat');
        return {
          actions: [{ type: 'continue', result: 'Rate limited, will retry' }],
          rawResponse: '',
          cost: { input_tokens: 0, output_tokens: 0, usd: 0 },
        };
      }

      if (err.status === 400 && err.message?.includes('SAFETY')) {
        logger.warn({ err }, 'Gemini safety filter triggered');
        return {
          actions: [
            {
              type: 'blocked',
              reason: 'Content blocked by Gemini safety filters',
            },
          ],
          rawResponse: '',
          cost: { input_tokens: 0, output_tokens: 0, usd: 0 },
        };
      }

      throw err;
    }
  }
}

TypeScript

import type { AgentRuntime } from './types.js';
import { ClaudeRuntime } from './claude.js';
import { OpenAIRuntime } from './openai.js';
import { GeminiRuntime } from './gemini.js';         // ← NOWE
import { env } from '../env.js';
import { logger } from '../utils/logger.js';

const runtimes: Map<string, AgentRuntime> = new Map();

export function initRuntimes() {
  if (env.ANTHROPIC_API_KEY) {
    runtimes.set('claude', new ClaudeRuntime());
    logger.info('Registered Claude runtime');
  }

  if (env.OPENAI_API_KEY) {
    runtimes.set('openai', new OpenAIRuntime());
    logger.info('Registered OpenAI runtime');
  }

  if (env.GOOGLE_API_KEY) {                           // ← NOWE
    runtimes.set('gemini', new GeminiRuntime());       // ← NOWE
    logger.info('Registered Gemini runtime');           // ← NOWE
  }                                                    // ← NOWE

  if (runtimes.size === 0) {
    logger.warn('No API keys configured — no runtimes available');
  }
}

export function getRuntime(name: string): AgentRuntime {
  const runtime = runtimes.get(name);
  if (!runtime) {
    throw new Error(
      `Runtime "${name}" not available. Configured: ${[...runtimes.keys()].join(', ')}`
    );
  }
  return runtime;
}

export function listRuntimes(): string[] {
  return [...runtimes.keys()];
}
6. Zmiana w packages/server/src/db/schema.sql
Zmień constraint w tabeli agents — dodaj 'gemini' do dozwolonych runtime'ów.

Jeśli masz check constraint na kolumnie runtime, zmień go:

SQL

-- Jeśli masz CHECK constraint, zaktualizuj:
ALTER TABLE agents DROP CONSTRAINT IF EXISTS agents_runtime_check;
ALTER TABLE agents ADD CONSTRAINT agents_runtime_check
  CHECK (runtime IN ('claude', 'openai', 'gemini', 'http', 'bash'));
Jeśli nie masz check constrainta na runtime (kolumna to po prostu TEXT) — nic nie musisz zmieniać w schema. Gemini zadziała od razu.

TypeScript

To nie jest blokujące (Zod .string() też przepuści "gemini"), ale walidacja enum daje lepszy error message.
