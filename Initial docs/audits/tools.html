Architektura — jak tools działają

Schemat przepływu od heartbeatu do wykonania toola:

flow
Worker tick
  → processAgentHeartbeat(agentId)
      → buildAgentContext(agentId, taskId)   ← tu dodajemy AVAILABLE TOOLS
          → query agent_tools JOIN tools
          → formatToolForPrompt() dla każdego
          → wstrzyknięcie do additional_context
      → runtime.execute(context)             ← LLM wie o toolach
          → Claude zwraca action: use_tool
      → switch action.type
          case 'use_tool':
            → canUseTool(agentId, toolName)  ← sprawdź uprawnienia
            → evaluatePolicy(...)            ← sprawdź policies
            → executeTool(agentId, taskId, toolName, params)
                → switch tool.type
                    builtin → handleBuiltin()
                    http    → executeHttp()
                    bash    → execPromise() z whitelistą
                    mcp     → MCPService.callTool()
                → INSERT INTO tool_calls     ← log każdego wywołania

Typy tools obsługiwane przez system:

builtin — natywne funkcje (web_search, send_email…) http — zewnętrzne REST API bash — shell z whitelistą komend mcp — Model Context Protocol servers

import { exec } from 'child_process';
import { promisify } from 'util';
import { db } from '../db/client.js';
import { logger } from '../utils/logger.js';
import { MCPService } from '../services/mcp.js';

const execPromise = promisify(exec);

// ─── Bash helpers ─────────────────────────────────────────────────────────────

function getAllowedBashCommands(config: any): string[] {
  const cmds = config?.allowed_commands ?? config?.allowedCommands;
  if (!Array.isArray(cmds)) return [];
  return cmds.filter((v): v is string => typeof v === 'string')
             .map((c) => c.trim()).filter(Boolean);
}

function validateBashCommand(tool: any, params: any): string {
  const command = typeof params?.command === 'string' ? params.command.trim() : '';
  if (!command) throw new Error('Bash tool requires a non-empty command');
  const allowed = getAllowedBashCommands(tool.config);
  if (allowed.length === 0)
    throw new Error(`Bash tool "${tool.name}" is missing allowed_commands`);
  const ok = allowed.some((a) => command === a || command.startsWith(`${a} `));
  if (!ok) throw new Error(`Command not in whitelist: "${command}"`);
  return command;
}

// ─── HTTP helper ──────────────────────────────────────────────────────────────

async function executeHttp(tool: any, params: any): Promise<any> {
  const { url: baseUrl, method: defaultMethod = 'GET',
          headers: defaultHeaders = {}, timeout_ms = 15000 } = tool.config;
  const url = params.url ?? baseUrl;
  const method = (params.method ?? defaultMethod).toUpperCase();
  const headers = { 'Content-Type': 'application/json', ...defaultHeaders, ...params.headers };
  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), timeout_ms);
  try {
    const res = await fetch(url, {
      method, headers,
      body: ['GET', 'HEAD'].includes(method) ? undefined : JSON.stringify(params.body ?? params.data),
      signal: controller.signal,
    });
    clearTimeout(timer);
    const ct = res.headers.get('content-type') ?? '';
    const body = ct.includes('application/json') ? await res.json() : await res.text();
    if (!res.ok) throw new Error(`HTTP ${res.status}: ${JSON.stringify(body).slice(0, 200)}`);
    return { status: res.status, body };
  } finally { clearTimeout(timer); }
}

// ─── Builtin implementations ──────────────────────────────────────────────────

async function handleBuiltin(tool: any, params: any): Promise<any> {
  switch (tool.name) {

    case 'web_search': {
      const query = params.query;
      if (!query) throw new Error('web_search requires params.query');
      const apiKey = tool.config?.api_key ?? process.env.BRAVE_API_KEY;
      if (!apiKey) {
        logger.warn({ tool: tool.name }, 'No API key — returning mock results');
        return { query, results: [{ title: `Mock: ${query}`, url: 'https://example.com',
          snippet: 'Set BRAVE_API_KEY in tool config for real results.' }] };
      }
      const res = await fetch(`https://api.search.brave.com/res/v1/web/search?q=${encodeURIComponent(query)}&count=${params.count ?? 5}`,
        { headers: { 'Accept': 'application/json', 'X-Subscription-Token': apiKey } });
      if (!res.ok) throw new Error(`Brave Search error: ${res.status}`);
      const data = await res.json();
      return { query, results: (data.web?.results ?? []).map((r: any) =>
        ({ title: r.title, url: r.url, snippet: r.description })) };
    }

    case 'send_email': {
      const { to, subject, body } = params;
      if (!to || !subject || !body) throw new Error('send_email requires: to, subject, body');
      const apiKey = tool.config?.api_key ?? process.env.SENDGRID_API_KEY;
      const from = tool.config?.from ?? process.env.EMAIL_FROM;
      if (!apiKey || !from) throw new Error('send_email: missing api_key or from in config');
      const res = await fetch('https://api.sendgrid.com/v3/mail/send', {
        method: 'POST',
        headers: { Authorization: `Bearer ${apiKey}`, 'Content-Type': 'application/json' },
        body: JSON.stringify({ personalizations: [{ to: [{ email: to }] }],
          from: { email: from }, subject,
          content: [{ type: 'text/plain', value: body }] }),
      });
      if (!res.ok) throw new Error(`SendGrid error: ${res.status} ${await res.text()}`);
      return { ok: true, to, subject };
    }

    case 'create_document': {
      const { title, content, format = 'markdown' } = params;
      if (!title || !content) throw new Error('create_document requires: title, content');
      const result = await db.query(
        `INSERT INTO documents (company_id, title, content, format, created_by_tool)
         VALUES ((SELECT company_id FROM tools WHERE id = $1), $2, $3, $4, true)
         RETURNING id, title`,
        [tool.id, title, content, format]
      ).catch(() => null);
      return result
        ? { ok: true, document_id: result.rows[0]?.id, title }
        : { ok: true, title, content_length: content.length, note: 'documents table not yet created' };
    }

    case 'read_knowledge': {
      const { query, limit = 5 } = params;
      if (!query) throw new Error('read_knowledge requires params.query');
      const res = await db.query(
        `SELECT title, content, source FROM knowledge_items
         WHERE company_id = (SELECT company_id FROM tools WHERE id = $1)
           AND (title ILIKE $2 OR content ILIKE $2) LIMIT $3`,
        [tool.id, `%${query}%`, limit]
      );
      return { query, results: res.rows };
    }

    case 'wait': {
      const { reason = 'Waiting for external event', until } = params;
      return { ok: true, reason, until, note: 'Agent will be re-invoked at next heartbeat.' };
    }

    default:
      throw new Error(`Unknown builtin: "${tool.name}". Register in executor.ts or use type "http".`);
  }
}

// ─── Main executor ────────────────────────────────────────────────────────────

export async function executeTool(agentId: string, taskId: string, toolName: string, params: any): Promise<any> {
  const startTime = Date.now();
  const toolRes = await db.query(
    `SELECT t.* FROM tools t
     WHERE t.name = $1 AND t.company_id = (SELECT company_id FROM agents WHERE id = $2)`,
    [toolName, agentId]
  );
  if (toolRes.rows.length === 0) throw new Error(`Tool "${toolName}" not found. Register it in Tools → Add Tool.`);

  const tool = toolRes.rows[0];
  logger.info({ agentId, toolName, toolType: tool.type }, 'Executing tool');

  let output: any;
  let status: 'success' | 'error' = 'success';

  try {
    switch (tool.type) {
      case 'builtin': output = await handleBuiltin(tool, params); break;
      case 'bash': {
        const command = validateBashCommand(tool, params);
        const { stdout, stderr } = await execPromise(command, {
          timeout: tool.config?.timeout_ms ?? 10_000, maxBuffer: 1024 * 1024 });
        output = { stdout: stdout.trim(), stderr: stderr.trim() }; break;
      }
      case 'http': output = await executeHttp(tool, params); break;
      case 'mcp':
        output = await MCPService.callTool(tool.config.serverName,
          { command: tool.config.command, args: tool.config.args, env: tool.config.env },
          toolName, params); break;
      default: throw new Error(`Unsupported tool type: "${tool.type}"`);
    }
  } catch (err: any) {
    status = 'error'; output = { error: err.message };
    logger.error({ err, toolName, agentId }, 'Tool execution failed');
  }

  const duration_ms = Date.now() - startTime;
  await db.query(
    `INSERT INTO tool_calls (agent_id, task_id, tool_id, input, output, status, duration_ms)
     VALUES ($1, $2, $3, $4, $5, $6, $7)`,
    [agentId, taskId, tool.id, JSON.stringify(params), JSON.stringify(output), status, duration_ms]
  );
  if (status === 'error') throw new Error(output.error);
  return output;
}

function formatToolForPrompt(tool: { name: string; description: string | null; type: string; config: any }): string {
  const lines = [`- **${tool.name}** (type: ${tool.type})`, `  ${tool.description ?? 'No description.'}`];
  const paramHints: Record<string, string> = {
    web_search: '{ "query": "search term", "count": 5 }',
    send_email: '{ "to": "user@example.com", "subject": "...", "body": "..." }',
    create_document: '{ "title": "...", "content": "...", "format": "markdown" }',
    read_knowledge: '{ "query": "topic", "limit": 5 }',
    wait: '{ "reason": "Waiting...", "until": "2025-01-01T00:00:00Z" }',
  };
  switch (tool.type) {
    case 'bash':
      const allowed = tool.config?.allowed_commands ?? [];
      if (allowed.length) lines.push(`  Allowed commands: ${allowed.join(', ')}`);
      lines.push(`  Usage: { "type": "use_tool", "tool_name": "${tool.name}", "params": { "command": "ls -la" } }`);
      break;
    case 'http':
      lines.push(`  Calls: ${tool.config?.url ?? 'configured URL'} [${tool.config?.method ?? 'GET'}]`);
      lines.push(`  Usage: { "type": "use_tool", "tool_name": "${tool.name}", "params": { "body": {} } }`);
      break;
    case 'builtin':
      const hint = paramHints[tool.name];
      if (hint) lines.push(`  Params: ${hint}`);
      lines.push(`  Usage: { "type": "use_tool", "tool_name": "${tool.name}", "params": ${hint ?? '{}'} }`);
      break;
    case 'mcp':
      lines.push(`  MCP Server: ${tool.config?.serverName ?? 'configured'}`);
      lines.push(`  Usage: { "type": "use_tool", "tool_name": "${tool.name}", "params": { ... } }`);
      break;
  }
  return lines.join('\n');
}

function buildToolsContext(tools: Array<{ name: string; description: string | null; type: string; config: any }>): string {
  if (tools.length === 0) return `AVAILABLE TOOLS:\nNo tools assigned. You can only complete tasks, delegate, or send messages.`;
  return [
    `AVAILABLE TOOLS (${tools.length} total):`,
    `To use a tool, emit: { "type": "use_tool", "tool_name": "<name>", "params": { ... } }`,
    '',
    ...tools.map(formatToolForPrompt),
    '',
    `TOOL USAGE RULES:`,
    `- Always check tool results before proceeding`,
    `- If a tool returns an error, try an alternative or request_approval`,
    `- Do not call the same tool with identical params more than 3 times`,
    `- Prefer complete_task when you already have enough information`,
  ].join('\n');
}

export async function buildAgentContext(agentId: string, taskId: string): Promise<AgentContext> {
  // 1. Agent + Company
  const agentRes = await db.query(
    `SELECT a.*, c.name AS company_name, c.mission AS company_mission
     FROM agents a JOIN companies c ON a.company_id = c.id WHERE a.id = $1`, [agentId]);
  if (!agentRes.rows.length) throw new Error('Agent not found');
  const agent = agentRes.rows[0];

  // 2. Task
  const taskRes = await db.query('SELECT * FROM tasks WHERE id = $1', [taskId]);
  if (!taskRes.rows.length) throw new Error('Task not found');
  const task = taskRes.rows[0];

  // 3. Goal hierarchy
  const hierarchy: string[] = [];
  if (task.goal_id) {
    const r = await db.query(
      `WITH RECURSIVE goal_path AS (
         SELECT id, parent_id, title, 0 AS depth FROM goals WHERE id = $1
         UNION ALL
         SELECT g.id, g.parent_id, g.title, gp.depth + 1
         FROM goals g JOIN goal_path gp ON gp.parent_id = g.id
       ) SELECT title FROM goal_path ORDER BY depth DESC`, [task.goal_id]);
    hierarchy.push(...r.rows.map((row: any) => row.title));
  }

  // 4. Message history (last 20)
  const messagesRes = await db.query(
    `SELECT * FROM messages WHERE task_id = $1 ORDER BY created_at DESC LIMIT 20`, [taskId]);
  const history = messagesRes.rows.reverse().map((m: any) => ({
    role: (m.from_agent === agentId ? 'assistant' : 'user') as 'assistant' | 'user',
    content: m.content, metadata: m.metadata,
  }));

  // 5. Knowledge base
  const knowledgeRes = await KnowledgeService.search(agent.company_id, task.description || task.title, 5,
    { agentId, taskId, consumer: 'agent_context' });
  const knowledge_context = knowledgeRes.length > 0
    ? `COMPANY KNOWLEDGE:\n${knowledgeRes.map((k: any) => `--- ${k.title} ---\n${k.content}`).join('\n')}`
    : undefined;

  // 6. ── KLUCZOWE: Pobierz dostępne tools agenta ──────────────────────────────
  const toolsRes = await db.query(
    `SELECT t.name, t.description, t.type, t.config
     FROM tools t
     JOIN agent_tools at ON at.tool_id = t.id
     WHERE at.agent_id = $1 AND at.can_execute = true
     ORDER BY t.type, t.name`, [agentId]);

  return {
    company_name: agent.company_name, company_mission: agent.company_mission,
    agent_name: agent.name, agent_role: agent.role, agent_model: agent.model,
    agent_system_prompt: agent.system_prompt, knowledge_context, goal_hierarchy: hierarchy,
    current_task: { title: task.title, description: task.description },
    history,
    additional_context: buildToolsContext(toolsRes.rows),  // ← nowe pole
  };
}

import { Router } from 'express';
import { db } from '../db/client.js';
import { z } from 'zod';
import { requireRole } from '../middleware/auth.js';

const router: Router = Router({ mergeParams: true });

const toolCreateSchema = z.object({
  name: z.string().min(1).max(100).regex(/^[a-zA-Z0-9_-]+$/, 'Alphanumeric + _ -'),
  description: z.string().max(500).optional(),
  type: z.enum(['builtin', 'http', 'bash', 'mcp']).default('builtin'),
  config: z.record(z.any()).optional().default({}),
});
const assignSchema = z.object({
  agent_id: z.string().uuid(),
  can_execute: z.boolean().default(true),
  config: z.record(z.any()).optional().default({}),
});

async function getToolOrFail(toolId: string, companyId: string) {
  const res = await db.query('SELECT * FROM tools WHERE id = $1 AND company_id = $2', [toolId, companyId]);
  return res.rows[0] ?? null;
}

// GET / — lista z metrykami
router.get('/', requireRole(['owner', 'admin', 'member', 'viewer']), async (req, res, next) => {
  try {
    const companyId = req.params.companyId ?? (req.query.company_id as string);
    if (!companyId) return res.status(400).json({ error: 'Missing company_id' });
    const result = await db.query(`
      SELECT t.*,
        COUNT(DISTINCT at.agent_id)::int AS agent_count,
        COUNT(DISTINCT tc.id)::int AS total_calls,
        MAX(tc.created_at) AS last_called_at,
        AVG(CASE WHEN tc.status = 'success' THEN 1.0 ELSE 0.0 END) AS success_rate
      FROM tools t
      LEFT JOIN agent_tools at ON at.tool_id = t.id
      LEFT JOIN tool_calls tc ON tc.tool_id = t.id
      WHERE t.company_id = $1
      GROUP BY t.id ORDER BY t.type, t.created_at ASC`, [companyId]);
    res.json(result.rows);
  } catch (err) { next(err); }
});

// POST / — utwórz
router.post('/', requireRole(['owner', 'admin']), async (req, res, next) => {
  try {
    const parsed = toolCreateSchema.safeParse(req.body);
    if (!parsed.success) return res.status(400).json({ error: parsed.error.flatten() });

    res.status(201).json(result.rows[0]);
  } catch (err: any) {
    if (err.code === '23505') return res.status(409).json({ error: `Tool "${req.body.name}" already exists.` });
    next(err);
  }
});

// GET /:id — szczegóły + agenci + ostatnie wywołania
router.get('/:id', requireRole(['owner', 'admin', 'member', 'viewer']), async (req, res, next) => {

                FROM tool_calls tc LEFT JOIN agents a ON a.id = tc.agent_id
                WHERE tc.tool_id = $1 ORDER BY tc.created_at DESC LIMIT 5`, [req.params.id]),
    ]);
    res.json({ ...tool, assigned_agents: agentsRes.rows, recent_calls: callsRes.rows });
  } catch (err) { next(err); }
});

// PATCH /:id
router.patch('/:id', requireRole(['owner', 'admin']), async (req, res, next) => {
  try {
    const tool = await getToolOrFail(req.params.id, req.params.companyId);
    if (!tool) return res.status(404).json({ error: 'Tool not found' });
    const allowed = ['description', 'type', 'config'];
    const updates: string[] = []; const values: any[] = []; let idx = 1;
    for (const [k, v] of Object.entries(req.body)) {
      if (allowed.includes(k) && v !== undefined) {
        updates.push(`${k} = $${idx++}`); values.push(k === 'config' ? JSON.stringify(v) : v);
      }
    }
    if (!updates.length) return res.json(tool);
    values.push(req.params.id);
    const r = await db.query(`UPDATE tools SET ${updates.join(', ')} WHERE id = $${idx} RETURNING *`, values);
    res.json(r.rows[0]);
  } catch (err) { next(err); }
});

       RETURNING *`,
      [agent_id, req.params.id, can_execute, JSON.stringify(config)]);
    res.json(result.rows[0]);
  } catch (err) { next(err); }
});

  try {
    const tool = await getToolOrFail(req.params.id, req.params.companyId);
    if (!tool) return res.status(404).json({ error: 'Tool not found' });
    const params = req.body.params ?? {}; const t0 = Date.now();
    try {
      let output: any;
      if (tool.type === 'http') {
        const r = await fetch(tool.config.url ?? params.url, {
          method: params.method ?? tool.config.method ?? 'GET',
          headers: { 'Content-Type': 'application/json', ...tool.config.headers },
          body: ['GET', 'HEAD'].includes((params.method ?? 'GET').toUpperCase()) ? undefined : JSON.stringify(params.body ?? {}),
        });
        output = { status: r.status, body: (await r.text()).slice(0, 2000) };
      } else output = { note: `${tool.type} tools must be tested via agent heartbeat.` };
      res.json({ ok: true, duration_ms: Date.now() - t0, output });
    } catch (err: any) { res.json({ ok: false, duration_ms: Date.now() - t0, error: err.message }); }
  } catch (err) { next(err); }
});

// GET /:id/calls — historia wywołań
router.get('/:id/calls', requireRole(['owner', 'admin', 'member']), async (req, res, next) => {
  try {
    const tool = await getToolOrFail(req.params.id, req.params.companyId);
    if (!tool) return res.status(404).json({ error: 'Tool not found' });
    const limit = Math.min(parseInt(req.query.limit as string ?? '50', 10), 200);
    const offset = parseInt(req.query.offset as string ?? '0', 10);
    const result = await db.query(
      `SELECT tc.*, a.name AS agent_name, t.title AS task_title
       FROM tool_calls tc LEFT JOIN agents a ON a.id = tc.agent_id LEFT JOIN tasks t ON t.id = tc.task_id
       WHERE tc.tool_id = $1 ORDER BY tc.created_at DESC LIMIT $2 OFFSET $3`,
      [req.params.id, limit, offset]);
    res.json({ calls: result.rows, limit, offset });
  } catch (err) { next(err); }
});

] as const;

export async function seedDefaultTools(companyId: string): Promise<void> {
  logger.info({ companyId }, 'Seeding default tools');
  let created = 0;
  for (const tool of DEFAULT_TOOLS) {
    try {
      await db.query(
        `INSERT INTO tools (company_id, name, description, type, config)
         VALUES ($1, $2, $3, $4, $5) ON CONFLICT (company_id, name) DO NOTHING`,
        [companyId, tool.name, tool.description, tool.type, JSON.stringify(tool.config)]);
      created++;
    } catch (err: any) { logger.warn({ tool: tool.name, err: err.message }, 'Skip'); }
  }
  logger.info({ companyId, created }, 'Seed complete');
}

const companyId = process.argv[2];
if (companyId) {
  seedDefaultTools(companyId)
    .then(() => { console.log(`✓ Seeded ${DEFAULT_TOOLS.length} tools`); process.exit(0); })
    .catch((e) => { console.error(e); process.exit(1); });
}
