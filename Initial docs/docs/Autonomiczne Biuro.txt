Autonomiczne Biuro — Uniwersalny Zarys Architektury
________________________________________
1. Filozofia systemu
text
Ty nie piszesz "chatbota z wieloma agentami".
Ty modelujesz FIRMĘ — z ludźmi, celami, hierarchią i zasadami.

Każdy element systemu odpowiada czemuś z prawdziwej firmy:

  Agent        → Pracownik
  Task         → Zlecenie / ticket
  Goal         → Cel biznesowy
  Tool         → Narzędzie pracy (GitHub, email, przeglądarka...)
  Org Chart    → Struktura organizacyjna
  Budget       → Budżet działu / osoby
  Approval     → Podpis przełożonego
  Audit Log    → Dziennik decyzji
  Heartbeat    → "Przychodzę do pracy i sprawdzam co mam robić"
  Board (Ty)   → Zarząd

3. Model danych
SQL
-- ============ FIRMA ============

CREATE TABLE companies (
  id          UUID PRIMARY KEY,
  name        TEXT NOT NULL,
  mission     TEXT,            -- "Zbuduj #1 AI note-taking app"
  created_at  TIMESTAMPTZ DEFAULT now()
);

-- ============ CELE ============

CREATE TABLE goals (
  id          UUID PRIMARY KEY,
  company_id  UUID REFERENCES companies(id),
  parent_id   UUID REFERENCES goals(id),  -- hierarchia celów
  title       TEXT NOT NULL,
  description TEXT,
  status      TEXT DEFAULT 'active',       -- active | achieved | abandoned
  created_at  TIMESTAMPTZ DEFAULT now()
);

-- Cel firmy → cel działu → cel projektu → task
-- Każdy task "wie" dlaczego istnieje

-- ============ AGENCI ============

CREATE TABLE agents (
  id           UUID PRIMARY KEY,
  company_id   UUID REFERENCES companies(id),
  name         TEXT NOT NULL,              -- "Alice the CTO"
  role         TEXT NOT NULL,              -- "cto", "developer", "marketer"
  title        TEXT,                       -- "Chief Technology Officer"
  reports_to   UUID REFERENCES agents(id), -- hierarchia
  runtime      TEXT NOT NULL,              -- "claude-code", "openai", "http", "bash"
  config       JSONB DEFAULT '{}',         -- model, prompts, params
  system_prompt TEXT,
  status       TEXT DEFAULT 'idle',        -- idle | working | paused | terminated
  created_at   TIMESTAMPTZ DEFAULT now()
);

-- ============ BUDŻETY ============

CREATE TABLE budgets (
  id           UUID PRIMARY KEY,
  agent_id     UUID REFERENCES agents(id),
  month        DATE NOT NULL,              -- 2025-07-01
  limit_usd    NUMERIC(10,2) NOT NULL,     -- max $50/miesiąc
  spent_usd    NUMERIC(10,4) DEFAULT 0,
  UNIQUE(agent_id, month)
);

-- ============ TASKI ============

CREATE TABLE tasks (
  id           UUID PRIMARY KEY,
  company_id   UUID REFERENCES companies(id),
  goal_id      UUID REFERENCES goals(id),  -- ← łącznik z "dlaczego"
  parent_id    UUID REFERENCES tasks(id),  -- subtaski
  title        TEXT NOT NULL,
  description  TEXT,
  assigned_to  UUID REFERENCES agents(id),
  created_by   UUID,                        -- agent lub 'board'
  status       TEXT DEFAULT 'backlog',
    -- backlog | assigned | in_progress | review | done | blocked | cancelled
  priority     INT DEFAULT 0,
  locked_by    UUID REFERENCES agents(id),  -- atomic checkout
  locked_at    TIMESTAMPTZ,
  result       TEXT,                         -- co agent wyprodukował
  created_at   TIMESTAMPTZ DEFAULT now(),
  updated_at   TIMESTAMPTZ DEFAULT now()
);

-- ============ NARZĘDZIA ============

CREATE TABLE tools (
  id           UUID PRIMARY KEY,
  company_id   UUID REFERENCES companies(id),
  name         TEXT NOT NULL,               -- "github", "send_email", "web_search"
  type         TEXT NOT NULL,               -- "http", "function", "mcp", "bash"
  config       JSONB DEFAULT '{}',          -- endpoint, auth, schema
  description  TEXT                         -- dla agenta: kiedy i jak używać
);

CREATE TABLE agent_tools (
  agent_id     UUID REFERENCES agents(id),
  tool_id      UUID REFERENCES tools(id),
  permissions  JSONB DEFAULT '{}',          -- {"read": true, "write": false}
  PRIMARY KEY (agent_id, tool_id)
);

-- ============ WIADOMOŚCI ============

CREATE TABLE messages (
  id           UUID PRIMARY KEY,
  company_id   UUID REFERENCES companies(id),
  task_id      UUID REFERENCES tasks(id),   -- w kontekście jakiego taska
  from_agent   UUID,                         -- NULL = board/human
  to_agent     UUID,                         -- NULL = broadcast
  content      TEXT NOT NULL,
  type         TEXT DEFAULT 'message',
    -- message | delegation | status_update | approval_request | tool_call | tool_result
  metadata     JSONB DEFAULT '{}',
  created_at   TIMESTAMPTZ DEFAULT now()
);

-- ============ AUDIT LOG ============

CREATE TABLE audit_log (
  id           UUID PRIMARY KEY,
  company_id   UUID REFERENCES companies(id),
  agent_id     UUID,
  action       TEXT NOT NULL,
    -- task.created | task.completed | tool.called | budget.exceeded
    -- agent.hired | agent.terminated | approval.requested | approval.granted
  entity_type  TEXT,                         -- 'task', 'agent', 'tool'
  entity_id    UUID,
  details      JSONB DEFAULT '{}',
  cost_usd     NUMERIC(10,6),
  created_at   TIMESTAMPTZ DEFAULT now()
);

-- ============ APPROVAL GATES ============

CREATE TABLE approvals (
  id           UUID PRIMARY KEY,
  company_id   UUID REFERENCES companies(id),
  requested_by UUID REFERENCES agents(id),
  type         TEXT NOT NULL,
    -- hire_agent | spend_over_threshold | deploy | strategy_change
  payload      JSONB NOT NULL,
  status       TEXT DEFAULT 'pending',       -- pending | approved | rejected
  decided_by   TEXT,                          -- 'board' lub agent_id
  decided_at   TIMESTAMPTZ,
  created_at   TIMESTAMPTZ DEFAULT now()
);

4. Orchestrator — serce systemu
text
┌─────────────────────────────────────────┐
│            ORCHESTRATOR LOOP            │
│                                         │
│  co N sekund (lub cron):                │
│                                         │
│  1. Dla każdego agenta:                 │
│     - Czy ma heartbeat do wykonania?    │
│     - Czy budżet pozwala?              │
│     - Czy ma przypisany task?          │
│                                         │
│  2. Jeśli tak:                         │
│     - Zbuduj KONTEKST                  │
│     - Wyślij do runtime'u agenta       │
│     - Odbierz wynik                    │
│     - Zapisz, zaloguj, rozlicz koszt   │
│                                         │
│  3. Obsłuż wynik:                      │
│     - Task done? → oznacz, powiadom    │
│     - Delegacja? → utwórz subtask     │
│     - Potrzebuje approval? → gate      │
│     - Chce użyć tool? → execute + log  │
│     - Chce gadać z innym agentem? → msg│
└─────────────────────────────────────────┘
Pseudokod orchestratora:
JavaScript
async function heartbeat(agent) {
  // 1. Sprawdź budżet
  const budget = await getBudget(agent.id, currentMonth());
  if (budget.spent_usd >= budget.limit_usd) {
    await log(agent, 'budget.exceeded');
    return; // nie pracuje
  }

  // 2. Znajdź robotę
  let task = await getActiveTask(agent.id);
  if (!task) {
    task = await checkoutNextTask(agent.id); // atomic lock!
  }
  if (!task) {
    await log(agent, 'heartbeat.idle');
    return; // nic do roboty
  }

  // 3. Zbuduj kontekst
  const context = await buildContext(agent, task);
  // context zawiera:
  //   - misja firmy
  //   - cel nadrzędny (goal ancestry)
  //   - opis taska + historia konwersacji
  //   - dostępne narzędzia
  //   - kto jest w zespole, kto jest przełożonym
  //   - poprzedni stan (jeśli sesja trwa)

  // 4. Wywołaj agenta
  const result = await executeAgent(agent, context);
  // result: { action, content, toolCalls, cost }

  // 5. Rozlicz koszt
  await addCost(agent.id, result.cost);

  // 6. Przetwórz wynik
  for (const action of result.actions) {
    switch (action.type) {
      case 'complete_task':
        await completeTask(task, action.result);
        break;
      case 'delegate':
        await createSubtask(task, action.to, action.description);
        break;
      case 'use_tool':
        await executeTool(agent, action.tool, action.params);
        break;
      case 'message':
        await sendMessage(agent, action.to, action.content, task);
        break;
      case 'request_approval':
        await createApproval(agent, action.payload);
        break;
      case 'blocked':
        await markBlocked(task, action.reason);
        break;
    }
  }

  // 7. Audit
  await log(agent, 'heartbeat.completed', { task, result });
}

5. Atomic Task Checkout
SQL
-- Agent "bierze" task — nikt inny nie może go wziąć jednocześnie

UPDATE tasks
SET
  locked_by = $agent_id,
  locked_at = now(),
  status = 'in_progress'
WHERE id = (
  SELECT id FROM tasks
  WHERE company_id = $company_id
    AND assigned_to = $agent_id
    AND status = 'assigned'
    AND locked_by IS NULL
  ORDER BY priority DESC, created_at ASC
  LIMIT 1
  FOR UPDATE SKIP LOCKED  -- ← klucz! pomija już zablokowane
)
RETURNING *;

6. Budowanie kontekstu dla agenta
JavaScript
async function buildContext(agent, task) {
  // Goal ancestry: task → goal → parent goal → mission
  const goalChain = await getGoalAncestry(task.goal_id);

  // Konwersacja na tym tasku
  const history = await getMessages(task.id);

  // Dostępne narzędzia
  const tools = await getAgentTools(agent.id);

  // Org context
  const boss = await getAgent(agent.reports_to);
  const reports = await getDirectReports(agent.id);

  // Poprzedni stan sesji (jeśli agent kontynuuje)
  const session = await getSession(agent.id, task.id);

  return {
    system: `
Jesteś ${agent.name}, ${agent.title} w firmie "${company.name}".
Misja firmy: ${company.mission}

Twoja rola: ${agent.role}
Raportujesz do: ${boss?.name || 'Board (zarząd)'}
Twoi podwładni: ${reports.map(r => r.name).join(', ') || 'brak'}

AKTUALNY CEL:
${goalChain.map((g, i) => '  '.repeat(i) + '→ ' + g.title).join('\n')}

AKTUALNY TASK:
Tytuł: ${task.title}
Opis: ${task.description}
Status: ${task.status}
Priorytet: ${task.priority}

DOSTĘPNE NARZĘDZIA:
${tools.map(t => `- ${t.name}: ${t.description}`).join('\n')}

DOSTĘPNE AKCJE:
- complete_task(result) — zakończ task z wynikiem
- delegate(to, description) — deleguj subtask do podwładnego
- use_tool(name, params) — użyj narzędzia
- message(to, content) — napisz do innego agenta
- request_approval(type, details) — poproś zarząd o zgodę
- blocked(reason) — zgłoś blokadę

HISTORIA KONWERSACJI:
${history.map(m => `[${m.from}]: ${m.content}`).join('\n')}
    `.trim(),
    session: session?.state || null,
  };
}

7. Tool Execution
JavaScript
const toolExecutors = {
  http: async (tool, params) => {
    const res = await fetch(tool.config.endpoint, {
      method: tool.config.method || 'POST',
      headers: { ...tool.config.headers, 'Content-Type': 'application/json' },
      body: JSON.stringify(params),
    });
    return res.json();
  },

  bash: async (tool, params) => {
    // UWAGA: sandbox! Nigdy nie exec na hoście bez izolacji
    return execInSandbox(tool.config.command, params);
  },

  mcp: async (tool, params) => {
    return mcpClient.call(tool.config.server, tool.name, params);
  },

  function: async (tool, params) => {
    const fn = registeredFunctions[tool.config.handler];
    return fn(params);
  },
};

async function executeTool(agent, toolName, params) {
  const tool = await getToolByName(agent.company_id, toolName);

  // Sprawdź uprawnienia
  const perms = await getToolPermissions(agent.id, tool.id);
  if (!perms) throw new Error(`Agent ${agent.name} nie ma dostępu do ${toolName}`);

  // Wykonaj
  const result = await toolExecutors[tool.type](tool, params);

  // Zaloguj
  await createMessage({
    task_id: currentTask.id,
    from_agent: agent.id,
    type: 'tool_call',
    content: `Called ${toolName}`,
    metadata: { params, result },
  });

  await audit({
    agent_id: agent.id,
    action: 'tool.called',
    entity_type: 'tool',
    entity_id: tool.id,
    details: { params, result },
  });

  return result;
}

8. Runaway Loop Detection
JavaScript
const LOOP_LIMITS = {
  maxHeartbeatsPerHour: 60,
  maxToolCallsPerTask: 100,
  maxDelegationDepth: 5,
  maxMessagesPerMinute: 20,
};

async function checkLoopSafety(agent, task) {
  const heartbeats = await countRecentHeartbeats(agent.id, '1 hour');
  if (heartbeats > LOOP_LIMITS.maxHeartbeatsPerHour) {
    await pauseAgent(agent.id, 'Too many heartbeats — possible loop');
    return false;
  }

  const toolCalls = await countToolCalls(task.id);
  if (toolCalls > LOOP_LIMITS.maxToolCallsPerTask) {
    await markBlocked(task, 'Too many tool calls — possible loop');
    return false;
  }

  const depth = await getDelegationDepth(task.id);
  if (depth > LOOP_LIMITS.maxDelegationDepth) {
    await markBlocked(task, 'Delegation too deep');
    return false;
  }

  return true;
}
